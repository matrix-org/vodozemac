28/// A unique identifier for a one-time [`Curve25519PublicKey`].
29///
30/// This identifier uses an internal counter to track the order in which
31/// one-time keys are generated.
32#[derive(Clone, Copy, Debug, Hash, PartialEq, Eq, PartialOrd, Ord, Serialize, Deserialize)]
33pub struct KeyId(pub(super) u64);
34
35impl From<KeyId> for String {
36    fn from(value: KeyId) -> String {
37        value.to_base64()
38    }
39}
40
41impl KeyId {
42    /// Encodes the [`KeyId`] in Base64.
43    ///
44    /// The internal counter is represented in big-endian format and then
45    /// encoded using Base64.
46    pub fn to_base64(self) -> String {
47        crate::utilities::base64_encode(self.0.to_be_bytes())
48    }
49}
50
51/// Error type for failures that may occur when decoding or using a
52/// cryptographic key.
53#[derive(Error, Debug)]
54pub enum KeyError {
55    /// Failed to correctly decode a public key that was encoded in Base64.
56    #[error("Failed to decode a public key from Base64: {0}")]
57    Base64Error(#[from] base64::DecodeError),
58
59    /// Failed to correctly decode a private key that was encoded in Base64.
60    #[error("Failed to decode a private key from Base64: {0}")]
61    Base64PrivateKey(#[from] base64ct::Error),
62
63    /// The Base64 encoded key does not contain the expected number of bytes.
64    #[error(
65        "Failed to decode {key_type} key from Base64: \
66        Invalid number of bytes for {key_type}, expected {expected_length}, got {length}."
67    )]
68    InvalidKeyLength {
69        /// The type of key being decoded.
70        key_type: &'static str,
71        /// The expected length of the key.
72        expected_length: usize,
73        /// The actual length of the key.
74        length: usize,
75    },
76    /// Unable to decompress the curve point `r` from an [`Ed25519PublicKey`].
77    ///
78    /// For more details, see [RFC 8032, Section 5.1.3](https://www.rfc-editor.org/rfc/rfc8032.html#section-5.1.3).
79    #[error(transparent)]
80    Signature(#[from] SignatureError),
81
82    /// One or more keys lacked contributory behavior in the Diffie-Hellman
83    /// operation, resulting in an insecure shared secret.
84    ///
85    /// For more details on contributory behavior please refer to the
86    /// [`x25519_dalek::SharedSecret::was_contributory()`] method.
87    #[error(
88        "One or more keys lacked contributory behavior in the Diffie-Hellman operation, \
89         resulting in an insecure shared secret"
90    )]
91    NonContributoryKey,
92}

94#[cfg(test)]
95mod test {
96    use crate::KeyId;
97
98    #[test]
99    fn key_id_to_base64() {
100        assert_eq!(KeyId(0).to_base64(), "AAAAAAAAAAA");
101        assert_eq!(KeyId(7).to_base64(), "AAAAAAAAAAc");
102    }
103
104    #[test]
105    fn key_id_to_string() {
106        assert_eq!(String::from(KeyId(0)), "AAAAAAAAAAA");
107        assert_eq!(String::from(KeyId(7)), "AAAAAAAAAAc");
108    }

15//! Implementation of an integrated encryption scheme.
16//!
17//! This module implements
18//! [ECIES](https://en.wikipedia.org/wiki/Integrated_Encryption_Scheme), the
19//! elliptic curve variant of the Integrated Encryption Scheme. This is a hybrid
20//! encryption scheme, using elliptic curve Diffie-Hellman for shared secret
21//! establishment and a symmetric algorithm for encryption of individual
22//! messages. It is instantiated with X25519 (Curve25519-based Diffie-Hellman),
23//! HMAC-SHA256 as the KDF and ChaCha20-Poly1305 for symmetric encryption.
24//!
25//! ECIES allows a party (the initiator) to establish a communication channel
26//! toward another party (the recipient) given knowledge of only its public key.
27//! We assume that this key was obtained in a secure way. This implies that the
28//! initiator side is able to tell for sure whether there is an active MITM
29//! attack in progress once the channel is established.
30//!
31//! On the other hand, the initiator's key pair is ephemeral and generated anew
32//! for each new channel. This implies the initiator must send their ephemeral
33//! public key to the recipient *unauthenticated* so that the recipient can
34//! complete the channel establishment on its end. From this it follows that the
35//! recipient has no way of knowing who is contacting them, allowing for active
36//! MITM attacks on the recipient side.
37//!
38//! In order to close this vector, an out-of-band confirmation is required to be
39//! sent from the initiator device to the recipient device, after which the
40//! channel is considered *secure*. The module provides the [`CheckCode`]
41//! facility which can be used for this purpose.
42//!
43//! Throughout this document, we use a naming convention which designates the
44//! device initiating an ECIES channel as device S, while the device on the
45//! other side (towards which the channel is opened) is designated device G.
46//!
47//! # Examples
48//!
49//! ```
50//! use vodozemac::ecies::{Ecies, InboundCreationResult, OutboundCreationResult};
51//!
52//! let plaintext = b"It's a secret to everybody";
53//!
54//! let alice = Ecies::new();
55//! let bob = Ecies::new();
56//!
57//! let OutboundCreationResult { ecies: mut alice, message } = alice
58//!     .establish_outbound_channel(bob.public_key(), plaintext)?;
59//!
60//! let InboundCreationResult { mut ecies, message } = bob
61//!     .establish_inbound_channel(&message)
62//!     .expect("We should be able to create an inbound channel");
63//!
64//! assert_eq!(
65//!     message, plaintext,
66//!     "The decrypted plaintext should match our initial plaintext"
67//! );
68//!
69//! // We now exchange the check code out-of-band and compare it.
70//! if alice.check_code() != ecies.check_code() {
71//!     panic!("The check code must match; possible active MITM attack in progress");
72//! }
73//!
74//! let message = ecies.encrypt(b"Another plaintext");
75//! let decrypted = alice.decrypt(&message)?;
76//!
77//! assert_eq!(decrypted, b"Another plaintext");
78//! # Ok::<(), anyhow::Error>(())
79//! ```

101/// The Error type for the ECIES submodule.
102#[derive(Debug, Error)]
103pub enum Error {
104    /// At least one of the keys did not have contributory behaviour and the
105    /// resulting shared secret would have been insecure.
106    #[error("At least one of the keys did not have contributory behaviour")]
107    NonContributoryKey,
108    /// Message decryption failed. Either the message was corrupted, the message
109    /// was replayed, or the wrong key is being used to decrypt the message.
110    #[error("Failed decrypting the message")]
111    Decryption,
112}

114/// A nonce that is used for the [`EstablishedEcies`] channel.
115///
116/// The nonce is internally represented as a [`u128`]. Each time a new value is
117/// retrieved, the counter will get incremented.
118struct EciesNonce {
119    inner: u128,
120}
121
122impl EciesNonce {
123    /// Create a new [`EciesNonce`], starting the count from 0.
124    const fn new() -> Self {
125        Self { inner: 0 }
126    }
127
128    /// Get the next nonce value.
129    ///
130    /// This will increment the underlying counter and return a 12 byte
131    /// [`Nonce`] value.
132    fn get(&mut self) -> Nonce {
133        let current = self.inner;
134        let (new_nonce, _) = self.inner.overflowing_add(1);
135        self.inner = new_nonce;
136
137        let mut nonce = [0u8; 12];
138        nonce.copy_from_slice(&current.to_le_bytes()[..12]);
139
140        #[allow(clippy::expect_used)]
141        #[allow(deprecated)]
142        Nonce::from_exact_iter(nonce)
143            .expect("We should be able to construct the correct nonce from a 12 byte slice")
144    }
145}

147/// A check code that can be used to confirm that two [`EstablishedEcies`]
148/// objects share the same secret. This is supposed to be shared out-of-band to
149/// protect against active MITM attacks.
150///
151/// Since the initiator device can always tell whether a MITM attack is in
152/// progress after channel establishment, this code technically carries only a
153/// single bit of information, representing whether the initiator has determined
154/// that the channel is "secure" or "not secure".
155///
156/// However, given this will need to be interactively confirmed by the user,
157/// there is risk that the user would confirm the dialogue without paying
158/// attention to its content. By expanding this single bit into a deterministic
159/// two-digit check code, the user is forced to pay more attention by having to
160/// enter it instead of just clicking through a dialogue.
161#[derive(Debug, Clone, PartialEq, Eq)]
162pub struct CheckCode {
163    bytes: [u8; 2],
164}
165
166impl CheckCode {
167    /// Convert the check code to an array of two bytes.
168    ///
169    /// The bytes can be converted to a more user-friendly representation. The
170    /// [`CheckCode::to_digit`] converts the bytes to a two-digit number.
171    pub const fn as_bytes(&self) -> &[u8; 2] {
172        &self.bytes
173    }
174
175    /// Convert the check code to two base-10 numbers.
176    ///
177    /// The number should be displayed with a leading 0 in case the first digit
178    /// is a 0.
179    ///
180    /// # Examples
181    ///
182    /// ```no_run
183    /// # use vodozemac::ecies::CheckCode;
184    /// # let check_code: CheckCode = unimplemented!();
185    /// let check_code = check_code.to_digit();
186    ///
187    /// println!("The check code of the IECS channel is: {check_code:02}");
188    /// ```
189    pub const fn to_digit(&self) -> u8 {
190        let first = (self.bytes[0] % 10) * 10;
191        let second = self.bytes[1] % 10;
192
193        first + second
194    }
195}

241    /// Create a new, random, unestablished ECIES session with the given
242    /// application info.
243    ///
244    /// The application info will be used to derive the various secrets and
245    /// provide domain separation.
246    pub fn with_info(info: &str) -> Self {
247        let rng = thread_rng();
248        let secret_key = EphemeralSecret::random_from_rng(rng);
249        let application_info_prefix = info.to_owned();
250
251        Self { secret_key, application_info_prefix }
252    }

254    /// Create an [`EstablishedEcies`] session using the other side's Curve25519
255    /// public key and an initial plaintext.
256    ///
257    /// After the channel has been established, we can encrypt messages to send
258    /// to the other side. The other side uses the initial message to
259    /// establishes the same channel on its side.
260    pub fn establish_outbound_channel(
261        self,
262        their_public_key: Curve25519PublicKey,
263        initial_plaintext: &[u8],
264    ) -> Result<OutboundCreationResult, Error> {
265        let our_public_key = self.public_key();
266        let shared_secret = self.secret_key.diffie_hellman(&their_public_key.inner);
267
268        if shared_secret.was_contributory() {
269            let mut ecies = EstablishedEcies::new(
270                &shared_secret,
271                our_public_key,
272                their_public_key,
273                &self.application_info_prefix,
274                Role::Initiator,
275            );
276
277            let message = ecies.encrypt(initial_plaintext);
278            let message =
279                InitialMessage { public_key: our_public_key, ciphertext: message.ciphertext };
280
281            Ok(OutboundCreationResult { ecies, message })
282        } else {
283            Err(Error::NonContributoryKey)
284        }
285    }
286
287    /// Create a [`EstablishedEcies`] from an [`InitialMessage`] encrypted by
288    /// the other side.
289    pub fn establish_inbound_channel(
290        self,
291        message: &InitialMessage,
292    ) -> Result<InboundCreationResult, Error> {
293        let our_public_key = self.public_key();
294
295        let shared_secret = self.secret_key.diffie_hellman(&message.public_key.inner);
296
297        if shared_secret.was_contributory() {
298            let mut ecies = EstablishedEcies::new(
299                &shared_secret,
300                our_public_key,
301                message.public_key,
302                &self.application_info_prefix,
303                Role::Recipient,
304            );
305
306            let nonce = ecies.decryption_nonce.get();
307            let message = ecies.decrypt_helper(&nonce, &message.ciphertext)?;
308
309            Ok(InboundCreationResult { ecies, message })
310        } else {
311            Err(Error::NonContributoryKey)
312        }
313    }

315    /// Get our [`Curve25519PublicKey`].
316    ///
317    /// This public key needs to be sent to the other side to be able to
318    /// establish an ECIES channel.
319    pub fn public_key(&self) -> Curve25519PublicKey {
320        Curve25519PublicKey::from(&self.secret_key)
321    }
322}

324/// An established ECIES session.
325///
326/// This session can be used to encrypt and decrypt messages between the two
327/// sides of the channel.
328#[derive(Zeroize, ZeroizeOnDrop)]
329pub struct EstablishedEcies {
330    /// Our own Curve25519 public key which was used to establish the ECIES
331    /// channel.
332    #[zeroize(skip)]
333    our_public_key: Curve25519PublicKey,
334
335    /// The other side's Curve25519 public key which was used to establish the
336    /// ECIES channel.
337    #[zeroize(skip)]
338    their_public_key: Curve25519PublicKey,
339
340    /// A counter which we'll use to create a [`Nonce`] every time we want to
341    /// encrypt a message.
342    #[zeroize(skip)]
343    encryption_nonce: EciesNonce,
344
345    /// A counter which we'll use to create a [`Nonce`] every time we want to
346    /// decrypt a message. The other side uses an analogous counter to encrypt
347    /// messages.
348    #[zeroize(skip)]
349    decryption_nonce: EciesNonce,
350
351    /// The key used to encrypt our messages.
352    encryption_key: Box<[u8; 32]>,
353
354    /// The key used by the other party to encrypt messages.
355    decryption_key: Box<[u8; 32]>,
356
357    /// The check code, generated on both devices and shared out-of-band, which
358    /// needs to match to ensure both sides are using the same secret.
359    #[zeroize(skip)]
360    check_code: CheckCode,
361
362    /// Our device's role in the ECIES channel, i.e. are we the initiator
363    /// (device S) or the recipient (device G)?
364    #[zeroize(skip)]
365    role: Role,
366}

368impl std::fmt::Debug for EstablishedEcies {
369    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
370        f.debug_struct("EstablishedEcies")
371            .field("our_public_key", &self.our_public_key)
372            .field("their_public_key", &self.their_public_key)
373            .field("check_code", &self.check_code)
374            .field("role", &self.role)
375            .finish()
376    }
377}

379impl EstablishedEcies {
380    fn create_check_code(
381        shared_secret: &SharedSecret,
382        our_public_key: Curve25519PublicKey,
383        their_public_key: Curve25519PublicKey,
384        info: &str,
385        role: Role,
386    ) -> CheckCode {
387        let mut bytes = [0u8; 2];
388        let kdf: Hkdf<Sha512> = Hkdf::new(None, shared_secret.as_bytes());
389
390        let info = Self::get_check_code_info(info, role, our_public_key, their_public_key);
391
392        #[allow(clippy::expect_used)]
393        kdf.expand(info.as_bytes(), bytes.as_mut_slice()).expect(
394            "We should be able to expand the 32-byte long shared secret into a 32 byte key.",
395        );
396
397        CheckCode { bytes }
398    }
399
400    fn create_key(info: &str, shared_secret: &SharedSecret) -> Box<[u8; 32]> {
401        let mut key = Box::new([0u8; 32]);
402        let kdf: Hkdf<Sha512> = Hkdf::new(None, shared_secret.as_bytes());
403
404        #[allow(clippy::expect_used)]
405        kdf.expand(info.as_bytes(), key.as_mut_slice()).expect(
406            "We should be able to expand the 32-byte long shared secret into a 32 byte key.",
407        );
408
409        key
410    }
411
412    /// Create the encryption key for messages we send into the channel.
413    fn create_encryption_key(
414        shared_secret: &SharedSecret,
415        our_public_key: Curve25519PublicKey,
416        their_public_key: Curve25519PublicKey,
417        app_info: &str,
418        role: Role,
419    ) -> Box<[u8; 32]> {
420        let info = Self::get_encryption_key_info(app_info, role, our_public_key, their_public_key);
421        Self::create_key(&info, shared_secret)
422    }
423
424    /// Create the decryption key for messages received from the other side of
425    /// the channel.
426    ///
427    /// The decryption key for G is the encryption key for S and vice versa.
428    fn create_decryption_key(
429        shared_secret: &SharedSecret,
430        our_public_key: Curve25519PublicKey,
431        their_public_key: Curve25519PublicKey,
432        app_info: &str,
433        role: Role,
434    ) -> Box<[u8; 32]> {
435        let info = Self::get_decryption_key_info(app_info, role, our_public_key, their_public_key);
436        Self::create_key(&info, shared_secret)
437    }
438
439    fn new(
440        shared_secret: &SharedSecret,
441        our_public_key: Curve25519PublicKey,
442        their_public_key: Curve25519PublicKey,
443        app_info: &str,
444        role: Role,
445    ) -> Self {
446        let (encryption_nonce, decryption_nonce) = (EciesNonce::new(), EciesNonce::new());
447
448        let encryption_key = Self::create_encryption_key(
449            shared_secret,
450            our_public_key,
451            their_public_key,
452            app_info,
453            role,
454        );
455        let decryption_key = Self::create_decryption_key(
456            shared_secret,
457            our_public_key,
458            their_public_key,
459            app_info,
460            role,
461        );
462        let check_code = Self::create_check_code(
463            shared_secret,
464            our_public_key,
465            their_public_key,
466            app_info,
467            role,
468        );
469
470        Self {
471            encryption_key,
472            decryption_key,
473            encryption_nonce,
474            decryption_nonce,
475            our_public_key,
476            their_public_key,
477            check_code,
478            role,
479        }
480    }
481
482    /// Get our [`Curve25519PublicKey`].
483    ///
484    /// This public key needs to be sent to the other side so that it can
485    /// complete the ECIES channel establishment.
486    pub const fn public_key(&self) -> Curve25519PublicKey {
487        self.our_public_key
488    }
489
490    /// Get the [`CheckCode`] which uniquely identifies this
491    /// [`EstablishedEcies`] session.
492    ///
493    /// This check code can be used to check that both sides of the session are
494    /// indeed using the same shared secret.
495    pub const fn check_code(&self) -> &CheckCode {
496        &self.check_code
497    }
498
499    fn encryption_key(&self) -> &Chacha20Key {
500        #[allow(deprecated)]
501        Chacha20Key::from_slice(self.encryption_key.as_slice())
502    }
503
504    fn decryption_key(&self) -> &Chacha20Key {
505        #[allow(deprecated)]
506        Chacha20Key::from_slice(self.decryption_key.as_slice())
507    }
508
509    /// Encrypt the given plaintext using this [`EstablishedEcies`] session.
510    pub fn encrypt(&mut self, plaintext: &[u8]) -> Message {
511        let nonce = self.encryption_nonce.get();
512
513        let cipher = ChaCha20Poly1305::new(self.encryption_key());
514        #[allow(clippy::expect_used)]
515        let ciphertext = cipher.encrypt(&nonce, plaintext).expect(
516            "We should always be able to encrypt a message since we provide the correct nonce",
517        );
518
519        Message { ciphertext }
520    }
521
522    /// Decrypt the given message using this [`EstablishedEcies`] session.
523    pub fn decrypt(&mut self, message: &Message) -> Result<Vec<u8>, Error> {
524        let nonce = self.decryption_nonce.get();
525        self.decrypt_helper(&nonce, &message.ciphertext)
526    }
527
528    fn decrypt_helper(&self, nonce: &Nonce, ciphertext: &[u8]) -> Result<Vec<u8>, Error> {
529        let cipher = ChaCha20Poly1305::new(self.decryption_key());
530        let plaintext = cipher.decrypt(nonce, ciphertext).map_err(|_| Error::Decryption)?;
531
532        Ok(plaintext)
533    }
534
535    fn get_check_code_info(
536        app_info: &str,
537        role: Role,
538        our_public_key: Curve25519PublicKey,
539        their_public_key: Curve25519PublicKey,
540    ) -> String {
541        let partial_info = format!("{app_info}_CHECKCODE");
542        Self::construct_info_string(&partial_info, role, our_public_key, their_public_key)
543    }
544
545    fn get_encryption_key_info(
546        app_info: &str,
547        role: Role,
548        our_public_key: Curve25519PublicKey,
549        their_public_key: Curve25519PublicKey,
550    ) -> String {
551        let partial_info = match role {
552            Role::Initiator => format!("{app_info}_ENCKEY_S"),
553            Role::Recipient => format!("{app_info}_ENCKEY_G"),
554        };
555        Self::construct_info_string(&partial_info, role, our_public_key, their_public_key)
556    }
557
558    fn get_decryption_key_info(
559        app_info: &str,
560        role: Role,
561        our_public_key: Curve25519PublicKey,
562        their_public_key: Curve25519PublicKey,
563    ) -> String {
564        // The decryption key for G is the encryption key for S and vice versa.
565        let partial_info = match role {
566            Role::Initiator => format!("{app_info}_ENCKEY_G"),
567            Role::Recipient => format!("{app_info}_ENCKEY_S"),
568        };
569        Self::construct_info_string(&partial_info, role, our_public_key, their_public_key)
570    }
571
572    fn construct_info_string(
573        partial_info: &str,
574        role: Role,
575        our_public_key: Curve25519PublicKey,
576        their_public_key: Curve25519PublicKey,
577    ) -> String {
578        match role {
579            Role::Recipient => {
580                // we are Device G. Gp = our_public_key, Sp = their_public_key
581                format!(
582                    "{partial_info}|{}|{}",
583                    our_public_key.to_base64(),
584                    their_public_key.to_base64(),
585                )
586            }
587            Role::Initiator => {
588                // we are Device S. Gp = their_public_key, Sp = our_public_key
589                format!(
590                    "{partial_info}|{}|{}",
591                    their_public_key.to_base64(),
592                    our_public_key.to_base64(),
593                )
594            }
595        }
596    }

607    fn channel_creation() {
608        let plaintext = b"It's a secret to everybody";
609

611        let bob = Ecies::new();
612
613        let OutboundCreationResult { ecies: mut alice, message } = alice

615            .expect("We should be able to create an outbound channel");

617        let InboundCreationResult { ecies: mut bob, message } = bob

623            "The decrypted plaintext should match our initial plaintext"

626        assert_eq!(alice.check_code().to_digit(), bob.check_code().to_digit());
627
628        let message = bob.encrypt(b"Another plaintext");
629
630        let decrypted =
631            alice.decrypt(&message).expect("We should be able to decrypt the second message");
632
633        assert_eq!(decrypted, b"Another plaintext");
634    }

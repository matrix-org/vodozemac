15//! Support for bidirectional communication in a HPKE channel.
16//!
17//! The scheme implemented here is described in the oblivous HTTP RFC in
18//! [section 4.4]. The one difference from the oblivious HTTP RFC is that we
19//! construct a HPKE context from the key and nonce we derive using the oblivous
20//! HTTP scheme instead of directly encrypting a message.
21//!
22//! [section 4.4]: https://www.rfc-editor.org/rfc/rfc9458#name-encapsulation-of-responses
23
24use chacha20poly1305::ChaCha20Poly1305;
25use cipher::{KeySizeUser, typenum::Unsigned};
26use hkdf::Hkdf;
27use hpke::{
28    HpkeError,
29    aead::{AeadCtxR, AeadCtxS, AeadKey, AeadNonce},
30    danger::streaming_enc::{ExporterSecret, create_receiver_context, create_sender_context},
31};
32use sha2::Sha256;
33use zeroize::Zeroize;
34
35use super::{Aead, Kdf, Kem};
36use crate::Curve25519PublicKey;

38pub(super) type AeadKeySize = <ChaCha20Poly1305 as KeySizeUser>::KeySize;

40/// A trait representing the creation of the paired context necessary to convert
41/// the inherently unidirectional HPKE channel into a bidirectional channel.
42///
43/// So for example, the original sender context gets a corresponding *receiver*
44/// context, allowing it to open (receive, decrypt) the receiver's responses.
45///
46/// Looking at it from the other side, the original receiver context gets a
47/// corresponding *sender* context, allowing it to seal (encrypt, send)
48/// responses to the original sender.
49pub(super) trait CreateResponseContext {
50    type ResponseContext;
51
52    fn export(&self, info: &[u8], output: &mut [u8]) -> Result<(), HpkeError>;
53
54    fn create_context(
55        &self,
56        response_key: &AeadKey<Aead>,
57        response_nonce: AeadNonce<Aead>,
58    ) -> Self::ResponseContext;
59
60    fn create_response_context(
61        &self,
62        application_info_prefix: &str,
63        encapsulated_key: Curve25519PublicKey,
64        response_nonce: &[u8],
65    ) -> Self::ResponseContext {
66        let mut secret = [0u8; AeadKeySize::USIZE];
67
68        // Export a secret from the HPKE context, we use our application info prefix and
69        // append "_RESPONSE" to it.
70        let info = format!("{application_info_prefix}_RESPONSE");
71
72        #[allow(clippy::expect_used)]
73        self.export(info.as_bytes(), &mut secret)
74            .expect("We should be able to export 32 bytes from the HPKE export interface");
75
76        // For the salt we concatenate the public key of the sender and the randomly
77        // generated response nonce.
78        let salt: Vec<u8> = [encapsulated_key.as_bytes().as_slice(), response_nonce].concat();
79
80        // Now create a KDF from the salt and the previously secret exported from the
81        // HPKE context.
82        let hkdf = Hkdf::<Sha256>::new(Some(&salt), &secret);
83
84        // From the KDF expand an AEAD key and nonce.
85        let mut aead_key = AeadKey::default();
86        let mut aead_nonce = AeadNonce::default();
87
88        #[allow(clippy::expect_used)]
89        hkdf.expand(b"key", aead_key.0.as_mut_slice())
90            .expect("We should be able to expand the base response secret into an AEAD key");
91
92        #[allow(clippy::expect_used)]
93        hkdf.expand(b"nonce", aead_nonce.0.as_mut_slice())
94            .expect("We should be able to expand the base response secret into a response nonce");
95
96        // Check that our key and nonce aren't just zeroes, this is only checked in
97        // debug builds.
98        debug_assert_ne!(aead_nonce.0.as_slice(), [0u8; 12]);
99        debug_assert_ne!(aead_key.0.as_slice(), [0u8; 32]);
100
101        // Let's get rid of the secret.
102        secret.zeroize();
103
104        // Now create a HPKE context which can be used to communicate in the other
105        // direction.
106        self.create_context(&aead_key, aead_nonce)
107    }
108}

110impl CreateResponseContext for AeadCtxS<Aead, Kdf, Kem> {
111    type ResponseContext = AeadCtxR<Aead, Kdf, Kem>;
112
113    fn export(&self, info: &[u8], output: &mut [u8]) -> Result<(), HpkeError> {
114        self.export(info, output)
115    }
116
117    fn create_context(
118        &self,
119        response_key: &AeadKey<Aead>,
120        response_nonce: AeadNonce<Aead>,
121    ) -> Self::ResponseContext {
122        // Response contexts (unlike sender contexts) will never export anything
123        // so we don't need an exporter secret. And yet the hpke crate's
124        // `create_ROLE_context` functions require it and we need to provide
125        // some value, so we provide a dummy default value. This is okay because
126        // our API doesn't expose an `export` method so this dummy exporter
127        // secret can never be accessed.
128        let exporter_secret = ExporterSecret::default();
129        create_receiver_context(response_key, response_nonce, exporter_secret)
130    }
131}

15//! Implementation of a [hybrid public encryption scheme].
16//!
17//! [hybrid public encryption scheme]: https://www.rfc-editor.org/rfc/rfc9180.html
18//!
19//! # Examples
20//!
21//! ```
22//! use vodozemac::hpke::*;
23//!
24//! let plaintext = b"It's a secret to everybody";
25//!
26//! let alice = HpkeSenderChannel::new();
27//! let bob = HpkeRecipientChannel::new();
28//!
29//! let SenderCreationResult { channel: mut alice, message } = alice
30//!     .establish_channel(bob.public_key(), plaintext, &[])?;
31//!
32//! let RecipientCreationResult { channel: mut bob, message } = bob.establish_channel(&message, &[])?;
33//!
34//! assert_eq!(
35//!     message, plaintext,
36//!     "The decrypted plaintext should match our initial plaintext"
37//! );
38//!
39//! // Now we need to establish communication in the other direction, Bob
40//! // needs to encrypt and send an initial reply to Alice.
41//!
42//! let plaintext = b"Not a secret to me!";
43//!
44//! let BidirectionalCreationResult { channel: mut bob, message } = bob.establish_bidirectional_channel(plaintext, &[]);
45//! let BidirectionalCreationResult { channel: mut alice, message } = alice.establish_bidirectional_channel(&message, &[])?;
46//!
47//! assert_eq!(message, plaintext);
48//!
49//! // We now exchange the check code out-of-band and compare it.
50//! if alice.check_code() != bob.check_code() {
51//!     panic!("The check code must match; possible active MITM attack in progress");
52//! }
53//!
54//! let message = bob.seal(b"Another plaintext", &[]);
55//! let decrypted = alice.open(&message, &[])?;
56//!
57//! assert_eq!(decrypted, b"Another plaintext");
58//! # Ok::<(), anyhow::Error>(())
59//! ```

88type SenderContext = AeadCtxS<Aead, Kdf, Kem>;
89type RecipientContext = AeadCtxR<Aead, Kdf, Kem>;
90type SenderResponseContext = AeadCtxR<Aead, Kdf, Kem>;
91type RecipientResponseContext = AeadCtxS<Aead, Kdf, Kem>;

93/// The possible device roles for an HPKE channel, indicating whether the
94/// device is initiating the channel or receiving/responding as the other side
95/// of the initiation.
96enum Role {
97    /// The role representing the side that sent the initial message.
98    Sender {
99        /// The established HPKE sender context.
100        sender_context: SenderContext,
101        /// The HPKE response context enabling the sender to receive messages
102        /// from the recipient.
103        response_context: SenderResponseContext,
104    },
105    /// The role representing the side that received the initial message and
106    /// sent the initial response.
107    Recipient {
108        /// The established HPKE recipient context.
109        sender_context: RecipientContext,
110        /// The HPKE response context enabling the recipient to send messages
111        /// to the sender.
112        response_context: RecipientResponseContext,
113    },
114}
115
116impl std::fmt::Debug for Role {
117    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
118        match self {
119            Role::Sender { .. } => f.write_str("Sender"),
120            Role::Recipient { .. } => f.write_str("Recipient"),
121        }
122    }
123}
124
125impl Role {
126    fn construct_info_string(
127        &self,
128        partial_info: &str,
129        our_public_key: Curve25519PublicKey,
130        their_public_key: Curve25519PublicKey,
131    ) -> String {
132        match self {
133            Role::Recipient { .. } => {
134                // we are Device G. Gp = our_public_key, Sp = their_public_key
135                format!(
136                    "{partial_info}|{}|{}",
137                    our_public_key.to_base64(),
138                    their_public_key.to_base64(),
139                )
140            }
141            Role::Sender { .. } => {
142                // we are Device S. Gp = their_public_key, Sp = our_public_key
143                format!(
144                    "{partial_info}|{}|{}",
145                    their_public_key.to_base64(),
146                    our_public_key.to_base64(),
147                )
148            }
149        }
150    }
151
152    fn check_code_info(
153        &self,
154        app_info: &str,
155        our_public_key: Curve25519PublicKey,
156        their_public_key: Curve25519PublicKey,
157    ) -> String {
158        let partial_info = format!("{app_info}_CHECKCODE");
159        self.construct_info_string(&partial_info, our_public_key, their_public_key)
160    }
161
162    fn check_code(
163        &self,
164        app_info: &str,
165        our_public_key: Curve25519PublicKey,
166        their_public_key: Curve25519PublicKey,
167    ) -> CheckCode {
168        let mut bytes = [0u8; 2];
169        let info = self.check_code_info(app_info, our_public_key, their_public_key);
170
171        let ret = match self {
172            Role::Sender { sender_context, .. } => {
173                sender_context.export(info.as_bytes(), &mut bytes)
174            }
175            Role::Recipient { sender_context, .. } => {
176                sender_context.export(info.as_bytes(), &mut bytes)
177            }
178        };
179
180        #[allow(clippy::expect_used)]
181        ret.expect("We should be able to generate a check code, as it's just two bytes");
182
183        CheckCode { bytes }
184    }
185}

187struct UnidirectionalHkpeChannel<T> {
188    /// The established HPKE context.
189    sender_context: T,
190
191    /// The application prefix which will be used as the info string to derive
192    /// secrets.
193    application_info_prefix: String,
194
195    /// Our own Curve25519 public key which was used to establish the HPKE
196    /// channel.
197    our_public_key: Curve25519PublicKey,
198
199    /// The other side's Curve25519 public key which was used to establish the
200    /// HPKE channel.
201    their_public_key: Curve25519PublicKey,
202}
203
204/// The result of the creation of a bidirectional and fully established HPKE
205/// channel.
206pub struct BidirectionalCreationResult<T> {
207    /// The established HPKE channel.
208    pub channel: EstablishedHpkeChannel,
209    /// The plaintext of the initial message.
210    pub message: T,
211}
212
213/// A fully established HPKE channel.
214///
215/// This channel allows full bidirecional communication with the other side.
216pub struct EstablishedHpkeChannel {
217    /// Our own Curve25519 public key which was used to establish the HPKE
218    /// channel.
219    our_public_key: Curve25519PublicKey,
220
221    /// The other side's Curve25519 public key which was used to establish the
222    /// HPKE channel.
223    their_public_key: Curve25519PublicKey,
224
225    /// Our device's role in the HPKE channel, i.e. are we the initiator
226    /// (device S) or the recipient (device G)?
227    role: Role,
228
229    /// The check code, generated on both devices and shared out-of-band, which
230    /// needs to match to ensure both sides are using the same secret.
231    check_code: CheckCode,
232}
233
234impl std::fmt::Debug for EstablishedHpkeChannel {
235    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
236        f.debug_struct("EstablishedHpkeChannel")
237            .field("our_public_key", &self.our_public_key)
238            .field("their_public_key", &self.their_public_key)
239            .field("check_code", &self.check_code)
240            .field("role", &self.role)
241            .finish()
242    }
243}
244
245impl EstablishedHpkeChannel {
246    /// Get our [`Curve25519PublicKey`].
247    ///
248    /// This public key needs to be sent to the other side so that it can
249    /// complete the HPKE channel establishment.
250    pub const fn public_key(&self) -> Curve25519PublicKey {
251        self.our_public_key
252    }
253
254    /// Get the [`Curve25519PublicKey`] of the other participant in this
255    /// [`EstablishedHpkeChannel`].
256    pub const fn their_public_key(&self) -> Curve25519PublicKey {
257        self.their_public_key
258    }
259
260    /// Get the [`CheckCode`] which uniquely identifies this
261    /// [`EstablishedHpkeChannel`] session.
262    ///
263    /// This check code can be used to check that both sides of the session are
264    /// indeed using the same shared secret.
265    pub fn check_code(&self) -> &CheckCode {
266        &self.check_code
267    }
268
269    /// Seal the given plaintext using the associated data and this
270    /// [`EstablishedHpkeChannel`].
271    ///
272    /// This method will encrypt the given plaintext for the other side of this
273    /// [`EstablishedHpkeChannel`].
274    ///
275    /// # Panics
276    ///
277    /// If the message limit is reached. It's possible to seal 2^64 messages
278    /// with this channel. The other reason why this might panic if the
279    /// additional associated data is too big, it has to be shorter than
280    /// 2^64 bytes.
281    pub fn seal(&mut self, plaintext: &[u8], aad: &[u8]) -> Message {
282        let ret = match &mut self.role {
283            Role::Sender { sender_context, .. } => sender_context.seal(plaintext, aad),
284            Role::Recipient { response_context, .. } => response_context.seal(plaintext, aad),
285        };
286
287        #[allow(clippy::expect_used)]
288        let ciphertext = ret.expect(
289            "We should be able to seal a plaintext, unless we're overflowed the sequence counter",
290        );
291
292        Message { ciphertext }
293    }
294
295    /// Open the given message with the given additional associated data using
296    /// this [`EstablishedHpkeChannel`].
297    ///
298    /// This method will decrypt the given message which was encrypted using a
299    /// matching [`EstablishedHpkeChannel`].
300    pub fn open(&mut self, message: &Message, aad: &[u8]) -> Result<Vec<u8>, Error> {
301        let ret = match &mut self.role {
302            Role::Sender { response_context, .. } => {
303                response_context.open(&message.ciphertext, aad)
304            }
305            Role::Recipient { sender_context, .. } => sender_context.open(&message.ciphertext, aad),
306        };
307
308        ret.map_err(|_| Error::Decryption)
309    }
310}

319    #[test]
320    fn test_channel_creation() {
321        let alice = HpkeSenderChannel::new();
322        let bob = HpkeRecipientChannel::new();
323
324        let plaintext = b"It's a secret to everybody";
325
326        let SenderCreationResult { message, .. } = alice
327            .establish_channel(bob.public_key(), plaintext, &[])
328            .expect("We should be able to create the sender channel");
329
330        assert_ne!(message.ciphertext, plaintext);
331
332        let RecipientCreationResult { message, .. } = bob
333            .establish_channel(&message, &[])
334            .expect("We should be able to establish the recipient channel");
335
336        assert_eq!(message, plaintext);
337    }

339    #[test]
340    fn test_channel_roundtrip() {
341        let alice = HpkeSenderChannel::new();
342        let bob = HpkeRecipientChannel::new();
343
344        let bob_public_key = bob.public_key();
345
346        let plaintext = b"It's a secret to everybody";
347
348        let SenderCreationResult { channel: alice, message, .. } = alice
349            .establish_channel(bob_public_key, plaintext, &[])
350            .expect("We should be able to create the sender channel");
351
352        assert_ne!(message.ciphertext, plaintext);
353
354        let RecipientCreationResult { channel: bob, message } = bob
355            .establish_channel(&message, &[])
356            .expect("We should be able to establish the recipient channel");
357
358        assert_eq!(message, plaintext);
359
360        let plaintext = b"Not a secret to me!";
361
362        let BidirectionalCreationResult { message: initial_response, channel: mut bob } =
363            bob.establish_bidirectional_channel(plaintext, &[]);
364        assert_ne!(initial_response.ciphertext, plaintext);
365
366        let BidirectionalCreationResult { message: decrypted, channel: mut alice } = alice
367            .establish_bidirectional_channel(&initial_response, &[])
368            .expect("We should be able to decrypt the initial response");
369
370        assert_eq!(decrypted, plaintext);
371        assert_eq!(
372            alice.check_code(),
373            bob.check_code(),
374            "Alice and Bob should derive the same check code"
375        );
376
377        assert_eq!(
378            bob_public_key,
379            bob.public_key(),
380            "The public key should stay the same even after the bidirectional channel has been established"
381        );
382
383        assert_eq!(bob.their_public_key(), alice.public_key());
384
385        // Further messages can also be exchanged.
386        let plaintext = b"Fully";
387        let message = bob.seal(plaintext, &[]);
388        let decrypted =
389            alice.open(&message, &[]).expect("Alice should be able to open Bob's latest message");
390
391        assert_eq!(plaintext.as_slice(), decrypted);
392
393        alice.open(&message, &[]).expect_err("Replaying a message should not be possible");
394
395        let message = alice.seal(plaintext, b"some additional data");
396        bob.open(&message, &[]).expect_err(
397            "Bob should not be able to decrypt a message without providing the same AAD",
398        );
399
400        let message = bob
401            .open(&message, b"some additional data")
402            .expect("Bob should be able to decrypt Alice's final message");
403
404        assert_eq!(message, plaintext);
405    }

428    fn test_info_construction() {
429        use crate::types::Curve25519Keypair;
430
431        let app_info = "foobar";
432        let our_public_key = Curve25519Keypair::new().public_key;
433        let their_public_key = Curve25519Keypair::new().public_key;
434
435        let alice = HpkeSenderChannel::new();
436        let bob = HpkeRecipientChannel::new();
437
438        let SenderCreationResult { channel: alice, message } = alice
439            .establish_channel(bob.public_key(), b"", &[])
440            .expect("We should be able to create the sender channel");
441
442        let RecipientCreationResult { channel: bob, message: _ } = bob
443            .establish_channel(&message, &[])
444            .expect("We should be able to establish the recipient channel");
445
446        let BidirectionalCreationResult { channel: bob, message: initial_response } =
447            bob.establish_bidirectional_channel(b"My response", &[]);
448
449        let BidirectionalCreationResult { channel: alice, .. } = alice
450            .establish_bidirectional_channel(&initial_response, &[])
451            .expect("We should be able to establish the bidirectional channel for Alice");
452
453        let check_code_info1 =
454            alice.role.check_code_info(app_info, our_public_key, their_public_key);
455        assert_eq!(
456            check_code_info1,
457            format!("foobar_CHECKCODE|{their_public_key}|{our_public_key}")
458        );
459
460        let check_code_info2 = bob.role.check_code_info(app_info, our_public_key, their_public_key);
461        assert_eq!(
462            check_code_info2,
463            format!("foobar_CHECKCODE|{our_public_key}|{their_public_key}")
464        );
465    }

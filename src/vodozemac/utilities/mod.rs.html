27const STANDARD_NO_PAD: GeneralPurpose = GeneralPurpose::new(
28    &alphabet::STANDARD,
29    general_purpose::NO_PAD
30        .with_decode_padding_mode(base64::engine::DecodePaddingMode::Indifferent),
31);

33/// Decode the input as base64 with no padding.
34pub fn base64_decode(input: impl AsRef<[u8]>) -> Result<Vec<u8>, DecodeError> {
35    STANDARD_NO_PAD.decode(input)
36}

38/// Encode the input as base64 with no padding.
39pub fn base64_encode(input: impl AsRef<[u8]>) -> String {
40    STANDARD_NO_PAD.encode(input)
41}

43pub(crate) fn unpickle<T: for<'b> serde::Deserialize<'b>>(
44    ciphertext: &str,
45    pickle_key: &[u8; 32],
46) -> Result<T, crate::PickleError> {
47    use zeroize::Zeroize;
48
49    let cipher = crate::cipher::Cipher::new_pickle(pickle_key);
50    let decoded = base64_decode(ciphertext)?;
51    let mut plaintext = cipher.decrypt_pickle(&decoded)?;
52
53    let pickle = serde_json::from_slice(&plaintext)?;
54
55    plaintext.zeroize();
56
57    Ok(pickle)
58}

60pub(crate) fn pickle<T: serde::Serialize>(thing: &T, pickle_key: &[u8; 32]) -> String {
61    use zeroize::Zeroize;
62
63    #[allow(clippy::expect_used)]
64    let mut json = serde_json::to_vec(&thing)
65        .expect("A pickled object should always be serializable into JSON");
66    let cipher = crate::cipher::Cipher::new_pickle(pickle_key);
67
68    let ciphertext = cipher.encrypt_pickle(json.as_slice());
69
70    json.zeroize();
71
72    base64_encode(ciphertext)
73}

75pub(crate) fn extract_mac(slice: &[u8], truncated: bool) -> crate::cipher::MessageMac {
76    use crate::cipher::Mac;
77
78    if truncated {
79        let mac_slice = &slice[0..Mac::TRUNCATED_LEN];
80
81        let mut mac = [0u8; Mac::TRUNCATED_LEN];
82        mac.copy_from_slice(mac_slice);
83        mac.into()
84    } else {
85        let mac_slice = &slice[0..Mac::LENGTH];
86
87        let mut mac = [0u8; Mac::LENGTH];
88        mac.copy_from_slice(mac_slice);
89        Mac(mac).into()
90    }
91}

103pub(crate) trait VarInt {
104    fn to_var_int(self) -> Vec<u8>;
105}
106
107/// Most-significant byte, == 0x80
108const MSB: u8 = 0b1000_0000;
109
110/// How many bytes an integer uses when being encoded as a VarInt.
111#[inline]
112const fn required_encoded_space_unsigned(mut v: u64) -> usize {
113    if v == 0 {
114        return 1;
115    }
116
117    let mut logcounter = 0;
118    while v > 0 {
119        logcounter += 1;
120        v >>= 7;
121    }
122    logcounter
123}
124
125impl VarInt for usize {
126    fn to_var_int(self) -> Vec<u8> {
127        (self as u64).to_var_int()
128    }
129}
130
131impl VarInt for u32 {
132    fn to_var_int(self) -> Vec<u8> {
133        (self as u64).to_var_int()
134    }
135}
136
137impl VarInt for u64 {
138    #[inline]
139    fn to_var_int(self) -> Vec<u8> {
140        let mut v = vec![0u8; required_encoded_space_unsigned(self)];
141
142        let mut n = self;
143        let mut i = 0;
144
145        while n >= 0x80 {
146            v[i] = MSB | (n as u8);
147            i += 1;
148            n >>= 7;
149        }
150
151        v[i] = n as u8;
152
153        v
154    }
155}

157#[cfg(test)]
158mod test {

160
161    use super::*;

163    #[test]
164    fn test_b64_decode_any_padding() {
165        let encoded_with_padding = "VGhpc0lzQVRlc3Q=";
166        let encoded_without_padding = "VGhpc0lzQVRlc3Q";
167
168        let first = base64_decode(encoded_with_padding).expect("Should decode if there is padding");
169        let second =
170            base64_decode(encoded_without_padding).expect("Should decode if there is no padding");
171
172        assert_eq!(
173            first, second,
174            "Decoding the same base64 string with and without padding should produce the same result"
175        )
176    }
177
178    #[test]
179    #[timeout(10)]
180    fn integer_encoding_required_space() {
181        assert_eq!(required_encoded_space_unsigned(0), 1);
182        assert_eq!(required_encoded_space_unsigned(100), 1);
183        assert_eq!(required_encoded_space_unsigned(1000), 2);
184        assert_eq!(required_encoded_space_unsigned(10000), 2);
185        assert_eq!(required_encoded_space_unsigned(100000), 3);
186        assert_eq!(required_encoded_space_unsigned(1000000), 3);
187        assert_eq!(required_encoded_space_unsigned(10000000), 4);
188        assert_eq!(required_encoded_space_unsigned(1000000000), 5);
189    }

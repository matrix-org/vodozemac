24const ADVANCEMENT_SEEDS: [&[u8; 1]; Ratchet::RATCHET_PART_COUNT] =
25    [b"\x00", b"\x01", b"\x02", b"\x03"];
26
27#[derive(Clone, Serialize, Deserialize, Zeroize, ZeroizeOnDrop)]
28pub(super) struct Ratchet {
29    inner: RatchetBytes,
30    counter: u32,
31}
32
33impl ConstantTimeEq for Ratchet {
34    fn ct_eq(&self, other: &Self) -> Choice {
35        // Short circuit on the counter, not on the contents of the ratchet.
36        if self.counter != other.counter {
37            Choice::from(0)
38        } else {
39            self.as_bytes().ct_eq(other.as_bytes())
40        }
41    }
42}

44#[derive(Clone, Zeroize, ZeroizeOnDrop)]
45struct RatchetBytes(Box<[u8; Ratchet::RATCHET_LENGTH]>);
46
47impl RatchetBytes {
48    fn from_bytes(bytes: &[u8]) -> Result<Self, RatchetBytesError> {
49        let length = bytes.len();
50
51        if length != Ratchet::RATCHET_LENGTH {
52            Err(RatchetBytesError::InvalidLength(length))
53        } else {
54            let mut ratchet = Self(Box::new([0u8; Ratchet::RATCHET_LENGTH]));
55            ratchet.0.copy_from_slice(bytes);
56
57            Ok(ratchet)
58        }
59    }
60}
61
62impl Serialize for RatchetBytes {
63    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
64    where
65        S: Serializer,
66    {
67        let bytes = &self.0;
68        bytes.serialize(serializer)
69    }
70}
71
72impl<'d> Deserialize<'d> for RatchetBytes {
73    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
74    where
75        D: Deserializer<'d>,
76    {
77        let mut bytes = <Vec<u8>>::deserialize(deserializer)?;
78        let ratchet = Self::from_bytes(bytes.as_ref()).map_err(serde::de::Error::custom)?;
79
80        bytes.zeroize();
81
82        Ok(ratchet)
83    }
84}

86struct RatchetPart<'a>(&'a mut [u8]);
87
88impl RatchetPart<'_> {
89    fn hash(&self, seed: &[u8; 1]) -> CtOutput<Hmac<Sha256>> {
90        #[allow(clippy::expect_used)]
91        let mut hmac = Hmac::<Sha256>::new_from_slice(self.0)
92            .expect("We should be able to create a HMAC object from a ratchet part");
93        hmac.update(seed);
94
95        hmac.finalize()
96    }
97
98    fn update(&mut self, new_part: &[u8]) {
99        self.0.copy_from_slice(new_part);
100    }
101}
102
103struct RatchetParts<'a> {
104    r_0: RatchetPart<'a>,
105    r_1: RatchetPart<'a>,
106    r_2: RatchetPart<'a>,
107    r_3: RatchetPart<'a>,
108}
109
110impl<'a> RatchetParts<'a> {
111    fn update(&'a mut self, from: usize, to: usize) {
112        #[allow(clippy::unreachable)]
113        let from = match from {
114            0 => &self.r_0,
115            1 => &self.r_1,
116            2 => &self.r_2,
117            3 => &self.r_3,
118            _ => unreachable!("We only have 4 ratchet parts"),
119        };
120
121        let result = from.hash(ADVANCEMENT_SEEDS[to]);
122
123        #[allow(clippy::unreachable)]
124        let to = match to {
125            0 => &mut self.r_0,
126            1 => &mut self.r_1,
127            2 => &mut self.r_2,
128            3 => &mut self.r_3,
129            _ => unreachable!("We only have 4 ratchet parts"),
130        };
131
132        to.update(&result.into_bytes());
133    }
134}

136impl Ratchet {
137    pub const RATCHET_LENGTH: usize = 128;
138    const RATCHET_PART_COUNT: usize = 4;
139    const LAST_RATCHET_INDEX: usize = Self::RATCHET_PART_COUNT - 1;
140
141    pub fn new() -> Self {
142        let mut rng = thread_rng();
143
144        let mut ratchet =
145            Self { inner: RatchetBytes(Box::new([0u8; Self::RATCHET_LENGTH])), counter: 0 };
146
147        rng.fill_bytes(ratchet.inner.0.as_mut());
148
149        ratchet
150    }
151
152    pub const fn from_bytes(bytes: Box<[u8; Self::RATCHET_LENGTH]>, counter: u32) -> Self {
153        Self { inner: RatchetBytes(bytes), counter }
154    }
155
156    pub const fn index(&self) -> u32 {
157        self.counter
158    }
159
160    pub const fn as_bytes(&self) -> &[u8; Self::RATCHET_LENGTH] {
161        &self.inner.0
162    }
163
164    #[cfg(test)]
165    pub const fn ratchet_bytes_pointer(&self) -> *const [u8; 128] {
166        &*self.inner.0
167    }
168
169    fn as_parts(&mut self) -> RatchetParts<'_> {
170        let (top, bottom) = self.inner.0.split_at_mut(64);
171
172        let (r_0, r_1) = top.split_at_mut(32);
173        let (r_2, r_3) = bottom.split_at_mut(32);
174
175        let r_0 = RatchetPart(r_0);
176        let r_1 = RatchetPart(r_1);
177        let r_2 = RatchetPart(r_2);
178        let r_3 = RatchetPart(r_3);
179
180        RatchetParts { r_0, r_1, r_2, r_3 }
181    }
182
183    pub fn advance(&mut self) {
184        let mut mask: u32 = 0x00FFFFFF;
185
186        // The index of the "slowest" part of the ratchet that needs to be
187        // advanced.
188        let mut h = 0;
189
190        self.counter += 1;
191
192        // Figure out which parts of the ratchet need to be advanced.
193        for i in 0..Self::RATCHET_LENGTH {
194            if (self.counter & mask) == 0 {
195                h = i;
196                break;
197            }
198
199            mask >>= 8;
200        }
201
202        let parts_to_advance = (h..=Self::LAST_RATCHET_INDEX).rev();
203
204        // Now advance R(h)...R(3) based on R(h).
205        for i in parts_to_advance {
206            let mut parts = self.as_parts();
207            parts.update(h, i);
208        }
209    }
210
211    pub fn advance_to(&mut self, advance_to: u32) {
212        for j in 0..Self::RATCHET_PART_COUNT {
213            let shift = (Self::LAST_RATCHET_INDEX - j) * 8;
214            let mask: u32 = !0u32 << shift;
215
216            // How many times do we need to rehash this part? `& 0xff` ensures
217            // we handle integer wrap-around correctly.
218            let mut steps = ((advance_to >> shift) - (self.counter >> shift)) & 0xff;
219
220            if steps == 0 {
221                // Deal with the edge case where the ratchet counter is slightly
222                // larger than the index we need to advance to. This should only
223                // happen for R(0) and implies that advance_to has wrapped
224                // around and we need to advance R(0) 256 times.
225                if advance_to < self.counter {
226                    steps = 0x100;
227                } else {
228                    continue;
229                }
230            }
231
232            // For all but the last step, we can just bump R(j) without regard
233            // to R(j+1)...R(3).
234            while steps > 1 {
235                let mut parts = self.as_parts();
236                parts.update(j, j);
237                steps -= 1;
238            }
239
240            // On the last step we also need to bump R(j+1)...R(3).
241            // (Theoretically, we could skip bumping R(j+2) if we're going to
242            // bump R(j+1) again, but the code to figure that out is a bit
243            // baroque and doesn't save us much).
244
245            let parts_to_update = (j..=Self::LAST_RATCHET_INDEX).rev();
246
247            for k in parts_to_update {
248                let mut parts = self.as_parts();
249                parts.update(j, k);
250            }
251
252            self.counter = advance_to & mask;
253        }
254    }
255}

330    #[test]
331    fn update_modifies_the_correct_part() {
332        let mut ratchet = Ratchet::from_bytes([0; 128].into(), 0);
333
334        {
335            let parts = ratchet.as_parts();
336
337            // At the beginning we're all zero bytes.
338            assert_eq!(parts.r_0.0, [0; 32]);
339            assert_eq!(parts.r_1.0, [0; 32]);
340            assert_eq!(parts.r_2.0, [0; 32]);
341            assert_eq!(parts.r_3.0, [0; 32]);
342        }
343
344        {
345            let mut parts = ratchet.as_parts();
346            parts.update(0, 1);
347        }
348
349        {
350            let parts = ratchet.as_parts();
351
352            // Now we advanced the zeroth part and put the result into the first part.
353            assert_eq!(parts.r_0.0, [0; 32]);
354            assert_ne!(parts.r_1.0, [0; 32]);
355            assert_eq!(parts.r_2.0, [0; 32]);
356            assert_eq!(parts.r_3.0, [0; 32]);
357        }
358
359        {
360            let mut parts = ratchet.as_parts();
361            parts.update(1, 2);
362        }
363
364        {
365            let parts = ratchet.as_parts();
366
367            // Now we advanced the third part and put the result into the second part.
368            assert_eq!(parts.r_0.0, [0; 32]);
369            assert_ne!(parts.r_1.0, [0; 32]);
370            assert_ne!(parts.r_2.0, [0; 32]);
371            assert_eq!(parts.r_3.0, [0; 32]);
372        }
373
374        {
375            let mut parts = ratchet.as_parts();
376            parts.update(2, 3);
377        }
378
379        {
380            let parts = ratchet.as_parts();
381
382            // Now we advanced the second part and put the result into the zeroth part.
383            assert_eq!(parts.r_0.0, [0; 32]);
384            assert_ne!(parts.r_1.0, [0; 32]);
385            assert_ne!(parts.r_2.0, [0; 32]);
386            assert_ne!(parts.r_3.0, [0; 32]);
387        }
388
389        {
390            let mut parts = ratchet.as_parts();
391            parts.update(3, 0);
392        }
393
394        {
395            let parts = ratchet.as_parts();
396
397            // Now we advanced the second part and put the result into the zeroth part.
398            assert_ne!(parts.r_0.0, [0; 32]);
399            assert_ne!(parts.r_1.0, [0; 32]);
400            assert_ne!(parts.r_2.0, [0; 32]);
401            assert_ne!(parts.r_3.0, [0; 32]);
402        }
403    }

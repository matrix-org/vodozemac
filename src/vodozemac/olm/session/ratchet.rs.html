27/// A ratchet key which we created ourselves.
28///
29/// A new ratchet key is created each time the conversation changes direction,
30/// and used to calculate the [root key](RootKey) for the new sender chain.
31/// The public part of the sender's ratchet key is sent to the recipient in each
32/// message.
33///
34/// Since this is *our own* key, we have both the secret and public parts of the
35/// key.
36///
37/// The [Olm spec](https://gitlab.matrix.org/matrix-org/olm/blob/master/docs/olm.md) refers to
38/// ratchet keys as `T`<sub>`i`</sub>.
39#[derive(Serialize, Deserialize, Clone)]
40#[serde(transparent)]
41pub(super) struct RatchetKey(Curve25519SecretKey);
42
43/// The public part of a [`RatchetKey`].
44#[derive(Debug, PartialEq, Eq, Clone, Copy)]
45pub struct RatchetPublicKey(Curve25519PublicKey);
46
47/// A ratchet key which was created by the other side.
48///
49/// See [`RatchetKey`] for explanation about ratchet keys in general. Since this
50/// is the other side's key, we have only the public part of the key.
51#[derive(Clone, Copy, Hash, PartialEq, Eq, Serialize, Deserialize, Decode)]
52#[serde(transparent)]
53pub struct RemoteRatchetKey(Curve25519PublicKey);
54
55impl Debug for RemoteRatchetKey {
56    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
57        self.0.fmt(f)
58    }
59}
60
61impl RatchetKey {
62    pub fn new() -> Self {
63        Self(Curve25519SecretKey::new())
64    }
65
66    pub fn diffie_hellman(&self, other: &RemoteRatchetKey) -> SharedSecret {
67        self.0.diffie_hellman(&other.0)
68    }
69}
70
71impl Default for RatchetKey {
72    fn default() -> Self {
73        Self::new()
74    }
75}

77impl From<Curve25519SecretKey> for RatchetKey {
78    fn from(key: Curve25519SecretKey) -> Self {
79        Self(key)
80    }
81}

95impl From<Curve25519PublicKey> for RemoteRatchetKey {

101impl AsRef<Curve25519PublicKey> for RatchetPublicKey {

113/// Information about the root key ratchet, while our sender chain is active.
114///
115/// We only have one of these while the double ratchet is "active" - ie, while
116/// we are encrypting messages. It stores the information necessary to calculate
117/// the *next* root key; in particular, the root key of our active chain
118/// `R`<sub>`i`</sub>, and our own ratchet key `T`<sub>`i`</sub>.
119#[derive(Serialize, Deserialize, Clone)]
120pub(super) struct Ratchet {
121    root_key: RootKey,
122    ratchet_key: RatchetKey,
123}
124
125impl Ratchet {
126    pub fn new(root_key: RootKey) -> Self {
127        let ratchet_key = RatchetKey::new();
128
129        Self { root_key, ratchet_key }
130    }
131
132    pub const fn new_with_ratchet_key(root_key: RootKey, ratchet_key: RatchetKey) -> Self {
133        Self { root_key, ratchet_key }
134    }
135
136    pub fn advance(&self, remote_key: RemoteRatchetKey) -> (RemoteRootKey, RemoteChainKey) {
137        let (remote_root_key, remote_chain_key) =
138            self.root_key.advance(&self.ratchet_key, &remote_key);
139
140        (remote_root_key, remote_chain_key)
141    }
142
143    pub const fn ratchet_key(&self) -> &RatchetKey {
144        &self.ratchet_key
145    }
146}

148#[cfg(test)]
149mod test {
150    use super::RatchetKey;
151    use crate::{Curve25519SecretKey, olm::RatchetPublicKey};
152
153    #[test]
154    fn ratchet_key_from_curve_25519_secret_key() {
155        let bytes = b"aaaaaaaaaaaaaaawaaaaaaaaaaaaaaaa";
156        let key = RatchetKey::from(Curve25519SecretKey::from_slice(bytes));
157        assert_eq!(key.0.to_bytes().as_ref(), bytes);
158    }
159
160    #[test]
161    fn ratchet_public_key_from_bytes() {
162        let bytes = b"aaaaaaaaaaaaaaawaaaaaaaaaaaaaaaa";
163        let key = RatchetPublicKey::from(*bytes);
164        assert_eq!(key.0.to_bytes().as_ref(), bytes);
165    }

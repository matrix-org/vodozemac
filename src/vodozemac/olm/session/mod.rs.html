55/// Error type for Olm-based decryption failures.
56#[derive(Error, Debug)]
57pub enum DecryptionError {
58    /// The message authentication code of the message was invalid.
59    #[error("Failed decrypting Olm message, invalid MAC: {0}")]
60    InvalidMAC(#[from] MacError),
61    /// The length of the message authentication code of the message did not
62    /// match our expected length.
63    #[error("Failed decrypting Olm message, invalid MAC length: expected {0}, got {1}")]
64    InvalidMACLength(usize, usize),
65    /// The ciphertext of the message isn't padded correctly.
66    #[error("Failed decrypting Olm message, invalid padding")]
67    InvalidPadding(#[from] UnpadError),
68    /// One or more keys lacked contributory behavior in the Diffie-Hellman
69    /// operation, resulting in an insecure shared secret.
70    ///
71    /// For more details on contributory behavior please refer to the
72    /// [`x25519_dalek::SharedSecret::was_contributory()`] method.
73    #[error(
74        "One or more keys lacked contributory behavior in the Diffie-Hellman operation, \
75         resulting in an insecure shared secret"
76    )]
77    NonContributoryKey,
78    /// The session is missing the correct message key to decrypt the message,
79    /// either because it was already used up, or because the Session has been
80    /// ratcheted forwards and the message key has been discarded.
81    #[error("The message key with the given key can't be created, message index: {0}")]
82    MissingMessageKey(u64),
83    /// Too many messages have been skipped to attempt decrypting this message.
84    #[error("The message gap was too big, got {0}, max allowed {1}")]
85    TooBigMessageGap(u64, u64),
86}

114        if self.inner.is_full() {
115            self.inner.pop_at(0);
116        }
117
118        self.inner.push(ratchet)
119    }

123    }

132        self.inner.get(index)
133    }

136        self.inner.iter_mut().find(|r| r.belongs_to(ratchet_key))

146/// An Olm session represents one end of an encrypted communication channel
147/// between two participants.
148///
149/// A session enables enables the session owner to encrypt messages intended
150/// for, and decrypt messages sent by, the other participant of the channel.
151///
152/// Olm sessions have two important properties:
153///
154/// 1. They are based on a double ratchet algorithm which continuously
155///    introduces new entropy into the channel as messages are sent and
156///    received. This imbues the channel with *self-healing* properties,
157///    allowing it to recover from a momentary loss of confidentiality in the
158///    event of a key compromise.
159/// 2. They are *asynchronous*, allowing the participant to start sending
160///    messages to the other side even if the other participant is not online at
161///    the moment.
162///
163/// An Olm [`Session`] is acquired from an [`Account`], by calling either
164///
165/// - [`Account::create_outbound_session`], if you are the first participant to
166///   send a message in this channel, or
167/// - [`Account::create_inbound_session`], if the other participant initiated
168///   the channel by sending you a message.
169///
170/// [`Account`]: crate::olm::Account
171/// [`Account::create_outbound_session`]: crate::olm::Account::create_outbound_session
172/// [`Account::create_inbound_session`]: crate::olm::Account::create_inbound_session
173pub struct Session {
174    session_keys: SessionKeys,
175    sending_ratchet: DoubleRatchet,
176    receiving_chains: ChainStore,
177    config: SessionConfig,
178}
179
180impl Debug for Session {
181    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
182        let Self { session_keys: _, sending_ratchet, receiving_chains, config } = self;
183
184        f.debug_struct("Session")
185            .field("session_id", &self.session_id())
186            .field("sending_ratchet", &sending_ratchet)
187            .field("receiving_chains", &receiving_chains.inner)
188            .field("config", config)
189            .finish_non_exhaustive()
190    }
191}

209    pub(super) fn new_remote(
210        config: SessionConfig,
211        shared_secret: RemoteShared3DHSecret,
212        remote_ratchet_key: Curve25519PublicKey,
213        session_keys: SessionKeys,
214    ) -> Self {
215        let (root_key, remote_chain_key) = shared_secret.expand();
216
217        let remote_ratchet_key = RemoteRatchetKey::from(remote_ratchet_key);
218        let root_key = RemoteRootKey::new(root_key);
219        let remote_chain_key = RemoteChainKey::new(remote_chain_key);
220
221        let local_ratchet = DoubleRatchet::inactive_from_prekey_data(root_key, remote_ratchet_key);
222        let remote_ratchet =
223            ReceiverChain::new(remote_ratchet_key, remote_chain_key, RatchetCount::new());
224
225        let mut ratchet_store = ChainStore::new();
226        ratchet_store.push(remote_ratchet);
227
228        Self {
229            session_keys,
230            sending_ratchet: local_ratchet,
231            receiving_chains: ratchet_store,
232            config,
233        }
234    }

243    /// Have we ever received and decrypted a message from the other side?

247    pub const fn has_received_message(&self) -> bool {

249    }

251    /// Encrypt the `plaintext` and construct an [`OlmMessage`].
252    ///
253    /// The message will either be a pre-key message or a normal message,
254    /// depending on whether the session is fully established. A [`Session`] is
255    /// fully established once you receive (and decrypt) at least one
256    /// message from the other side.
257    pub fn encrypt(&mut self, plaintext: impl AsRef<[u8]>) -> Result<OlmMessage, EncryptionError> {
258        let message = match self.config.version {
259            Version::V1 => self.sending_ratchet.encrypt_truncated_mac(plaintext.as_ref()),
260            #[cfg(feature = "experimental-session-config")]
261            Version::V2 => self.sending_ratchet.encrypt(plaintext.as_ref()),
262        }?;
263
264        if self.has_received_message() {
265            Ok(OlmMessage::Normal(message))
266        } else {
267            let message = PreKeyMessage::new(self.session_keys, message);
268
269            Ok(OlmMessage::PreKey(message))
270        }
271    }

283    /// Get the [`MessageKey`] to encrypt the next message.
284    ///
285    /// **Note**: Each key obtained in this way should be used to encrypt
286    /// a message and the message must then be sent to the recipient.
287    ///
288    /// Failing to do so will increase the number of out-of-order messages on
289    /// the recipient side. Given that a `Session` can only support a limited
290    /// number of out-of-order messages, this will eventually lead to
291    /// undecryptable messages.

294        self.sending_ratchet.next_message_key()
295    }

297    /// Try to decrypt an Olm message, which will either return the plaintext or
298    /// result in a [`DecryptionError`].
299    pub fn decrypt(&mut self, message: &OlmMessage) -> Result<Vec<u8>, DecryptionError> {
300        let decrypted = match message {
301            OlmMessage::Normal(m) => self.decrypt_decoded(m)?,
302            OlmMessage::PreKey(m) => self.decrypt_decoded(&m.message)?,
303        };
304
305        Ok(decrypted)
306    }
307
308    pub(super) fn decrypt_decoded(
309        &mut self,
310        message: &Message,
311    ) -> Result<Vec<u8>, DecryptionError> {
312        let ratchet_key = RemoteRatchetKey::from(message.ratchet_key);
313
314        if let Some(ratchet) = self.receiving_chains.find_ratchet(&ratchet_key) {
315            ratchet.decrypt(message, &self.config)
316        } else {
317            let (sending_ratchet, mut remote_ratchet) =
318                self.sending_ratchet
319                    .advance(ratchet_key)
320                    .ok_or(DecryptionError::NonContributoryKey)?;
321
322            let plaintext = remote_ratchet.decrypt(message, &self.config)?;
323
324            self.sending_ratchet = sending_ratchet;
325            self.receiving_chains.push(remote_ratchet);
326
327            Ok(plaintext)
328        }
329    }

442        fn drop(&mut self) {

450    impl TryFrom<Pickle> for Session {
451        type Error = crate::LibolmPickleError;
452
453        fn try_from(pickle: Pickle) -> Result<Self, Self::Error> {
454            let mut receiving_chains = ChainStore::new();
455
456            for chain in &pickle.receiver_chains {
457                receiving_chains.push(chain.into())
458            }
459
460            for key in &pickle.message_keys {
461                let ratchet_key =
462                    RemoteRatchetKey::from(Curve25519PublicKey::from(key.ratchet_key));
463
464                if let Some(receiving_chain) = receiving_chains.find_ratchet(&ratchet_key) {
465                    receiving_chain.insert_message_key(key.into())
466                }
467            }
468
469            if let Some(chain) = pickle.sender_chains.first() {
470                // XXX: Passing in secret array as value.
471                let ratchet_key = RatchetKey::from(Curve25519SecretKey::from_slice(
472                    chain.secret_ratchet_key.as_ref(),
473                ));
474                let chain_key =
475                    ChainKey::from_bytes_and_index(chain.chain_key.clone(), chain.chain_key_index);
476
477                let root_key = RootKey::new(pickle.root_key.clone());
478
479                let ratchet = Ratchet::new_with_ratchet_key(root_key, ratchet_key);
480                let sending_ratchet = DoubleRatchet::from_ratchet_and_chain_key(ratchet, chain_key);
481
482                Ok(Self {
483                    session_keys: pickle.session_keys,
484                    sending_ratchet,
485                    receiving_chains,
486                    config: SessionConfig::version_1(),
487                })
488            } else if let Some(chain) = receiving_chains.get(0) {
489                let sending_ratchet = DoubleRatchet::inactive_from_libolm_pickle(
490                    RemoteRootKey::new(pickle.root_key.clone()),
491                    chain.ratchet_key(),
492                );
493
494                Ok(Self {
495                    session_keys: pickle.session_keys,
496                    sending_ratchet,
497                    receiving_chains,
498                    config: SessionConfig::version_1(),
499                })
500            } else {
501                Err(crate::LibolmPickleError::InvalidSession)
502            }
503        }
504    }

572    /// Then, create a pair of sessions between the two.
573    pub fn session_and_libolm_pair() -> Result<(Account, OlmAccount, Session, OlmSession)> {
574        let alice = Account::new();
575        let bob = OlmAccount::new();
576        bob.generate_one_time_keys(1);
577
578        let one_time_key = bob
579            .parsed_one_time_keys()
580            .curve25519()
581            .values()
582            .next()
583            .cloned()
584            .expect("Couldn't find a one-time key");

586        let identity_keys = bob.parsed_identity_keys();
587        let curve25519_key = Curve25519PublicKey::from_base64(identity_keys.curve25519())?;
588        let one_time_key = Curve25519PublicKey::from_base64(&one_time_key)?;
589        let mut alice_session = alice
590            .create_outbound_session(SessionConfig::version_1(), curve25519_key, one_time_key)

596        bob.mark_keys_as_published();

599            let session =

601
602            Ok((alice, bob, alice_session, session))

604            bail!("Invalid message type");
605        }

802
803        assert_eq!(
804            session.decrypt(&message).expect("Should be able to decrypt re-encrypted message"),
805            plaintext.as_bytes()

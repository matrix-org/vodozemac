28/// The sender side of a double-ratchet implementation.
29///
30/// While we are encrypting messages, we are in the "active" state. Here we need
31/// to keep track of the latest chain key `C`<sub>`i`,`j`</sub> (so that we can
32/// advance to the next one), and also the current root key `R`<sub>`i`</sub>
33/// and our most recent ratchet key `T`<sub>`i`</sub> (so that we can calculate
34/// the *next* root key).
35///
36/// Once we receive a message, we transition to the "inactive" state. Since we
37/// don't handle decryption here (that's done in [`ReceiverChain`]), we don't
38/// need to keep track of the sender's chain key. All we need is enough state so
39/// that we can calculate the next root key once we start encrypting again:
40/// specifically, the public part of the other side's ratchet key
41/// `T`<sub>`i`</sub> which was sent to us in the message, and the remote root
42/// key `R`<sub>`i`</sub>.
43#[derive(Serialize, Deserialize, Clone)]
44#[serde(transparent)]
45pub(super) struct DoubleRatchet {
46    inner: DoubleRatchetState,
47}
48
49impl DoubleRatchet {
50    pub fn next_message_key(&mut self) -> MessageKey {
51        match &mut self.inner {
52            DoubleRatchetState::Inactive(ratchet) => {
53                let mut ratchet = ratchet.activate();
54
55                let message_key = ratchet.next_message_key();
56                self.inner = DoubleRatchetState::Active(ratchet);
57
58                message_key
59            }
60            DoubleRatchetState::Active(ratchet) => ratchet.next_message_key(),
61        }
62    }
63
64    pub fn encrypt(&mut self, plaintext: &[u8]) -> Message {
65        self.next_message_key().encrypt(plaintext)
66    }
67
68    pub fn encrypt_truncated_mac(&mut self, plaintext: &[u8]) -> Message {
69        self.next_message_key().encrypt_truncated_mac(plaintext)
70    }
71
72    /// Create a new `DoubleRatchet` instance, based on a newly-calculated
73    /// shared secret.
74    pub fn active(shared_secret: Shared3DHSecret) -> Self {
75        let (root_key, chain_key) = shared_secret.expand();
76
77        let root_key = RootKey::new(root_key);
78        let chain_key = ChainKey::new(chain_key);
79
80        let ratchet = ActiveDoubleRatchet {
81            parent_ratchet_key: None, // First chain in a session lacks parent ratchet key
82            ratchet_count: RatchetCount::new(),
83            active_ratchet: Ratchet::new(root_key),
84            symmetric_key_ratchet: chain_key,
85        };
86
87        Self { inner: ratchet.into() }
88    }
89
90    #[cfg(feature = "libolm-compat")]
91    pub fn from_ratchet_and_chain_key(ratchet: Ratchet, chain_key: ChainKey) -> Self {
92        Self {
93            inner: ActiveDoubleRatchet {
94                parent_ratchet_key: None, // libolm pickle did not record parent ratchet key
95                ratchet_count: RatchetCount::unknown(), // nor the ratchet count
96                active_ratchet: ratchet,
97                symmetric_key_ratchet: chain_key,
98            }
99            .into(),
100        }
101    }
102
103    pub fn inactive_from_prekey_data(
104        root_key: RemoteRootKey,
105        ratchet_key: RemoteRatchetKey,
106    ) -> Self {
107        let ratchet_count = RatchetCount::new();
108        let ratchet = InactiveDoubleRatchet { root_key, ratchet_key, ratchet_count };
109
110        Self { inner: ratchet.into() }
111    }
112
113    #[cfg(feature = "libolm-compat")]
114    pub fn inactive_from_libolm_pickle(
115        root_key: RemoteRootKey,
116        ratchet_key: RemoteRatchetKey,
117    ) -> Self {
118        let ratchet_count = RatchetCount::unknown();
119        let ratchet = InactiveDoubleRatchet { root_key, ratchet_key, ratchet_count };
120
121        Self { inner: ratchet.into() }
122    }
123
124    pub fn advance(&mut self, ratchet_key: RemoteRatchetKey) -> (DoubleRatchet, ReceiverChain) {
125        let (ratchet, receiver_chain) = match &self.inner {
126            DoubleRatchetState::Active(r) => r.advance(ratchet_key),
127            DoubleRatchetState::Inactive(r) => {
128                let ratchet = r.activate();
129                // Advancing an inactive ratchet shouldn't be possible since the
130                // other side did not yet receive our new ratchet key.
131                //
132                // This will likely end up in a decryption error but for
133                // consistency sake and avoiding the leakage of our internal
134                // state it's better to error out there.
135                let ret = ratchet.advance(ratchet_key);
136
137                self.inner = ratchet.into();
138
139                ret
140            }
141        };
142
143        (Self { inner: DoubleRatchetState::Inactive(ratchet) }, receiver_chain)
144    }
145}

147impl Debug for DoubleRatchet {
148    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
149        let mut dbg = f.debug_tuple("DoubleRatchet");
150        match &self.inner {
151            DoubleRatchetState::Inactive(r) => dbg.field(r),
152            DoubleRatchetState::Active(r) => dbg.field(r),
153        };
154        dbg.finish()
155    }
156}

159#[serde(rename_all = "snake_case")]
160#[serde(tag = "type")]
161enum DoubleRatchetState {
162    Inactive(InactiveDoubleRatchet),
163    Active(ActiveDoubleRatchet),
164}
165
166impl From<InactiveDoubleRatchet> for DoubleRatchetState {
167    fn from(r: InactiveDoubleRatchet) -> Self {
168        Self::Inactive(r)
169    }

172impl From<ActiveDoubleRatchet> for DoubleRatchetState {
173    fn from(r: ActiveDoubleRatchet) -> Self {
174        Self::Active(r)

178/// State of the sender-side ratchet when we have received a new chain from the
179/// other side, and have not yet created a new chain of our own.
180///
181/// See [`DoubleRatchet`] for more explanation.
182#[derive(Serialize, Deserialize, Clone)]
183struct InactiveDoubleRatchet {
184    root_key: RemoteRootKey,
185    ratchet_key: RemoteRatchetKey,
186
187    /// The number of times the ratchet has been advanced.
188    ///
189    /// If `root_key` contains root key `R`<sub>`i`</sub>, this is `i`.
190    ///
191    /// This is not required to implement the algorithm: it is maintained solely
192    /// for diagnostic output.
193    #[serde(default = "RatchetCount::unknown")]
194    ratchet_count: RatchetCount,
195}
196
197impl InactiveDoubleRatchet {
198    fn activate(&self) -> ActiveDoubleRatchet {
199        let (root_key, chain_key, ratchet_key) = self.root_key.advance(&self.ratchet_key);
200        let active_ratchet = Ratchet::new_with_ratchet_key(root_key, ratchet_key);
201
202        ActiveDoubleRatchet {
203            parent_ratchet_key: Some(self.ratchet_key),
204            ratchet_count: self.ratchet_count.advance(),
205            active_ratchet,
206            symmetric_key_ratchet: chain_key,
207        }
208    }
209}
210
211impl Debug for InactiveDoubleRatchet {
212    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
213        f.debug_struct("InactiveDoubleRatchet")
214            .field("ratchet_count", &self.ratchet_count)
215            .field("ratchet_key", &self.ratchet_key)
216            .finish_non_exhaustive()
217    }
218}

220/// State of the sender-side ratchet while we are in "encryption" mode: we are
221/// encrypting our own messages and have not yet received any messages which
222/// were created since we started this chain.
223///
224/// See [`DoubleRatchet`] for more explanation.
225#[derive(Serialize, Deserialize, Clone)]
226struct ActiveDoubleRatchet {
227    /// The other side's most recent ratchet key, which was used to calculate
228    /// the root key in `active_ratchet` and the chain key in
229    /// `symmetric_key_ratchet`.
230    ///
231    /// If `active_ratchet` contains root key `R`<sub>`i`</sub> and our own
232    /// ratchet key `T`<sub>`i`</sub>, this is `T`<sub>`i-1`</sub>.
233    ///
234    /// `None` means "unknown", either because this session has been restored
235    /// from a pickle which did not record the parent session key, or because
236    /// this is the first chain in the session.
237    ///
238    /// This is not required to implement the algorithm: it is maintained solely
239    /// for diagnostic output.
240    #[serde(default)]
241    parent_ratchet_key: Option<RemoteRatchetKey>,
242
243    /// The number of times the ratchet has been advanced.
244    ///
245    /// If `active_ratchet` contains root key `R`<sub>`i`</sub>, this is `i`.
246    #[serde(default = "RatchetCount::unknown")]
247    ratchet_count: RatchetCount,
248
249    active_ratchet: Ratchet,
250    symmetric_key_ratchet: ChainKey,
251}
252
253impl ActiveDoubleRatchet {
254    fn advance(&self, ratchet_key: RemoteRatchetKey) -> (InactiveDoubleRatchet, ReceiverChain) {
255        let (root_key, remote_chain) = self.active_ratchet.advance(ratchet_key);
256
257        let new_ratchet_count = self.ratchet_count.advance();
258        let ratchet = InactiveDoubleRatchet {
259            root_key,
260            ratchet_key,
261            ratchet_count: new_ratchet_count.clone(),
262        };
263        let receiver_chain = ReceiverChain::new(ratchet_key, remote_chain, new_ratchet_count);
264
265        (ratchet, receiver_chain)
266    }
267
268    fn ratchet_key(&self) -> RatchetPublicKey {
269        RatchetPublicKey::from(self.active_ratchet.ratchet_key())
270    }
271
272    fn next_message_key(&mut self) -> MessageKey {
273        self.symmetric_key_ratchet.create_message_key(self.ratchet_key())
274    }
275}
276
277impl Debug for ActiveDoubleRatchet {
278    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
279        let active_ratchet_public_key: RatchetPublicKey = self.active_ratchet.ratchet_key().into();
280        f.debug_struct("ActiveDoubleRatchet")
281            .field("ratchet_count", &self.ratchet_count)
282            .field("parent_ratchet_key", &self.parent_ratchet_key)
283            .field("ratchet_key", &active_ratchet_public_key)
284            .field("chain_index", &self.symmetric_key_ratchet.index())
285            .finish_non_exhaustive()
286    }
287}

289/// The number of times the ratchet has been advanced, `i`.
290///
291/// This starts at 0 for the first prekey messages from Alice to Bob,
292/// increments to 1 when Bob replies, and then increments each time the
293/// conversation changes direction.
294///
295/// It may be unknown, if the ratchet was restored from a pickle
296/// which didn't track it.
297#[derive(Serialize, Deserialize, Clone, PartialEq, Eq)]
298pub enum RatchetCount {
299    Known(u64),
300    Unknown(()),
301}
302
303impl RatchetCount {
304    pub const fn new() -> RatchetCount {
305        RatchetCount::Known(0)
306    }
307
308    pub const fn unknown() -> RatchetCount {
309        RatchetCount::Unknown(())
310    }
311
312    pub fn advance(&self) -> RatchetCount {
313        match self {
314            RatchetCount::Known(count) => RatchetCount::Known(count + 1),
315            RatchetCount::Unknown(_) => RatchetCount::Unknown(()),
316        }
317    }
318}
319
320impl Debug for RatchetCount {
321    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
322        match self {
323            RatchetCount::Known(count) => write!(f, "{count}"),
324            RatchetCount::Unknown(_) => write!(f, "<unknown>"),
325        }
326    }
327}

338    fn create_session_pair(alice: &Account, bob: &mut Account) -> (Session, Session) {
339        let bob_otks = bob.generate_one_time_keys(1);
340        let bob_otk = bob_otks.created.first().expect("Couldn't get a one-time-key for bob");
341        let bob_identity_key = bob.identity_keys().curve25519;
342        let mut alice_session =
343            alice.create_outbound_session(SessionConfig::version_1(), bob_identity_key, *bob_otk);
344
345        let message = "It's a secret to everybody";
346        let olm_message = alice_session.encrypt(message);
347        assert_matches!(olm_message, OlmMessage::PreKey(prekey_message));
348
349        let alice_identity_key = alice.identity_keys().curve25519;
350        let bob_session_creation_result = bob
351            .create_inbound_session(alice_identity_key, &prekey_message)
352            .expect("Unable to create inbound session");
353        assert_eq!(bob_session_creation_result.plaintext, message.as_bytes());
354        (alice_session, bob_session_creation_result.session)
355    }

357    fn assert_active_ratchet(sending_ratchet: &DoubleRatchet) -> &ActiveDoubleRatchet {
358        match &sending_ratchet.inner {
359            DoubleRatchetState::Inactive(_) => panic!("Not an active ratchet"),
360            DoubleRatchetState::Active(s) => s,
361        }
362    }
363
364    fn assert_inactive_ratchet(sending_ratchet: &DoubleRatchet) -> &InactiveDoubleRatchet {
365        match &sending_ratchet.inner {
366            DoubleRatchetState::Active(_) => panic!("Not an inactive ratchet"),
367            DoubleRatchetState::Inactive(s) => s,
368        }
369    }
370
371    #[test]
372    fn ratchet_counts() {
373        let (mut alice_session, mut bob_session) =
374            create_session_pair(&Account::new(), &mut Account::new());
375
376        // Both ratchets should start with count 0.
377        assert_eq!(
378            assert_active_ratchet(&alice_session.sending_ratchet).ratchet_count,
379            RatchetCount::Known(0)
380        );
381        assert_eq!(
382            assert_inactive_ratchet(&bob_session.sending_ratchet).ratchet_count,
383            RatchetCount::Known(0)
384        );
385
386        // Once Bob replies, the ratchets should bump to 1.
387        let olm_message = bob_session.encrypt("sssh");
388        alice_session.decrypt(&olm_message).expect("Alice could not decrypt message from Bob");
389        assert_eq!(
390            assert_inactive_ratchet(&alice_session.sending_ratchet).ratchet_count,
391            RatchetCount::Known(1)
392        );
393        assert_eq!(
394            assert_active_ratchet(&bob_session.sending_ratchet).ratchet_count,
395            RatchetCount::Known(1)
396        );
397
398        // Now Alice replies again.
399        let olm_message = alice_session.encrypt("sssh");
400        bob_session.decrypt(&olm_message).expect("Bob could not decrypt message from Alice");
401        assert_eq!(
402            assert_active_ratchet(&alice_session.sending_ratchet).ratchet_count,
403            RatchetCount::Known(2)
404        );
405        assert_eq!(
406            assert_inactive_ratchet(&bob_session.sending_ratchet).ratchet_count,
407            RatchetCount::Known(2)
408        );
409    }

435            RatchetCount::Known(1)
436        );
437        assert_eq!(
438            assert_active_ratchet(&bob_session.sending_ratchet).ratchet_count,

442        // Now Alice replies again.
443        let olm_message = alice_session.encrypt("sssh");

445        assert_eq!(
446            assert_active_ratchet(&alice_session.sending_ratchet).ratchet_count,

28/// The sender side of a double-ratchet implementation.
29///
30/// While we are encrypting messages, we are in the "active" state. Here we need
31/// to keep track of the latest chain key `C`<sub>`i`,`j`</sub> (so that we can
32/// advance to the next one), and also the current root key `R`<sub>`i`</sub>
33/// and our most recent ratchet key `T`<sub>`i`</sub> (so that we can calculate
34/// the *next* root key).
35///
36/// Once we receive a message, we transition to the "inactive" state. Since we
37/// don't handle decryption here (that's done in [`ReceiverChain`]), we don't
38/// need to keep track of the sender's chain key. All we need is enough state so
39/// that we can calculate the next root key once we start encrypting again:
40/// specifically, the public part of the other side's ratchet key
41/// `T`<sub>`i`</sub> which was sent to us in the message, and the remote root
42/// key `R`<sub>`i`</sub>.
43#[derive(Serialize, Deserialize, Clone)]
44#[serde(transparent)]
45pub(super) struct DoubleRatchet {
46    inner: DoubleRatchetState,
47}
48
49impl DoubleRatchet {
50    pub fn next_message_key(&mut self) -> Option<MessageKey> {
51        match &mut self.inner {
52            DoubleRatchetState::Inactive(ratchet) => {
53                let mut ratchet = ratchet.activate()?;
54
55                let message_key = ratchet.next_message_key();
56                self.inner = DoubleRatchetState::Active(ratchet);
57
58                Some(message_key)
59            }
60            DoubleRatchetState::Active(ratchet) => Some(ratchet.next_message_key()),
61        }
62    }
63
64    #[cfg(feature = "experimental-session-config")]
65    pub fn encrypt(&mut self, plaintext: &[u8]) -> Result<Message, EncryptionError> {
66        Ok(self.next_message_key().ok_or(EncryptionError::NonContributoryKey)?.encrypt(plaintext))
67    }
68
69    pub fn encrypt_truncated_mac(&mut self, plaintext: &[u8]) -> Result<Message, EncryptionError> {
70        Ok(self
71            .next_message_key()
72            .ok_or(EncryptionError::NonContributoryKey)?
73            .encrypt_truncated_mac(plaintext))
74    }
75
76    /// Create a new `DoubleRatchet` instance, based on a newly-calculated
77    /// shared secret.
78    pub fn active(shared_secret: Shared3DHSecret) -> Self {
79        let (root_key, chain_key) = shared_secret.expand();
80
81        let root_key = RootKey::new(root_key);
82        let chain_key = ChainKey::new(chain_key);
83
84        let ratchet = ActiveDoubleRatchet {
85            parent_ratchet_key: None, // First chain in a session lacks parent ratchet key
86            ratchet_count: RatchetCount::new(),
87            active_ratchet: Ratchet::new(root_key),
88            symmetric_key_ratchet: chain_key,
89        };
90
91        Self { inner: ratchet.into() }
92    }
93
94    #[cfg(feature = "libolm-compat")]
95    pub fn from_ratchet_and_chain_key(ratchet: Ratchet, chain_key: ChainKey) -> Self {
96        Self {
97            inner: ActiveDoubleRatchet {
98                parent_ratchet_key: None, // libolm pickle did not record parent ratchet key
99                ratchet_count: RatchetCount::unknown(), // nor the ratchet count
100                active_ratchet: ratchet,
101                symmetric_key_ratchet: chain_key,
102            }
103            .into(),
104        }
105    }
106
107    pub fn inactive_from_prekey_data(
108        root_key: RemoteRootKey,
109        ratchet_key: RemoteRatchetKey,
110    ) -> Self {
111        let ratchet_count = RatchetCount::new();
112        let ratchet = InactiveDoubleRatchet { root_key, ratchet_key, ratchet_count };
113
114        Self { inner: ratchet.into() }
115    }
116
117    #[cfg(feature = "libolm-compat")]
118    pub fn inactive_from_libolm_pickle(
119        root_key: RemoteRootKey,
120        ratchet_key: RemoteRatchetKey,
121    ) -> Self {
122        let ratchet_count = RatchetCount::unknown();
123        let ratchet = InactiveDoubleRatchet { root_key, ratchet_key, ratchet_count };
124
125        Self { inner: ratchet.into() }
126    }
127
128    pub fn advance(
129        &mut self,
130        ratchet_key: RemoteRatchetKey,
131    ) -> Option<(DoubleRatchet, ReceiverChain)> {
132        let (ratchet, receiver_chain) = match &self.inner {
133            DoubleRatchetState::Active(r) => r.advance(ratchet_key)?,
134            DoubleRatchetState::Inactive(r) => {
135                let ratchet = r.activate()?;
136                // Advancing an inactive ratchet shouldn't be possible since the
137                // other side did not yet receive our new ratchet key.
138                //
139                // This will likely end up in a decryption error but for
140                // consistency sake and avoiding the leakage of our internal
141                // state it's better to error out there.
142                let ret = ratchet.advance(ratchet_key)?;
143
144                self.inner = ratchet.into();
145
146                ret
147            }
148        };
149
150        Some((Self { inner: DoubleRatchetState::Inactive(ratchet) }, receiver_chain))
151    }
152}

158            DoubleRatchetState::Inactive(r) => dbg.field(r),
159            DoubleRatchetState::Active(r) => dbg.field(r),

168enum DoubleRatchetState {
169    Inactive(InactiveDoubleRatchet),
170    Active(ActiveDoubleRatchet),
171}

173impl From<InactiveDoubleRatchet> for DoubleRatchetState {
174    fn from(r: InactiveDoubleRatchet) -> Self {
175        Self::Inactive(r)
176    }

180    fn from(r: ActiveDoubleRatchet) -> Self {

185/// State of the sender-side ratchet when we have received a new chain from the
186/// other side, and have not yet created a new chain of our own.
187///
188/// See [`DoubleRatchet`] for more explanation.
189#[derive(Serialize, Deserialize, Clone)]
190struct InactiveDoubleRatchet {
191    root_key: RemoteRootKey,
192    ratchet_key: RemoteRatchetKey,
193
194    /// The number of times the ratchet has been advanced.
195    ///
196    /// If `root_key` contains root key `R`<sub>`i`</sub>, this is `i`.
197    ///
198    /// This is not required to implement the algorithm: it is maintained solely
199    /// for diagnostic output.
200    #[serde(default = "RatchetCount::unknown")]
201    ratchet_count: RatchetCount,
202}
203
204impl InactiveDoubleRatchet {
205    fn activate(&self) -> Option<ActiveDoubleRatchet> {
206        let (root_key, chain_key, ratchet_key) = self.root_key.advance(&self.ratchet_key)?;
207        let active_ratchet = Ratchet::new_with_ratchet_key(root_key, ratchet_key);
208
209        Some(ActiveDoubleRatchet {
210            parent_ratchet_key: Some(self.ratchet_key),
211            ratchet_count: self.ratchet_count.advance(),
212            active_ratchet,
213            symmetric_key_ratchet: chain_key,
214        })
215    }
216}
217
218impl Debug for InactiveDoubleRatchet {
219    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
220        f.debug_struct("InactiveDoubleRatchet")
221            .field("ratchet_count", &self.ratchet_count)
222            .field("ratchet_key", &self.ratchet_key)
223            .finish_non_exhaustive()
224    }
225}

227/// State of the sender-side ratchet while we are in "encryption" mode: we are
228/// encrypting our own messages and have not yet received any messages which
229/// were created since we started this chain.
230///
231/// See [`DoubleRatchet`] for more explanation.
232#[derive(Serialize, Deserialize, Clone)]
233struct ActiveDoubleRatchet {
234    /// The other side's most recent ratchet key, which was used to calculate
235    /// the root key in `active_ratchet` and the chain key in
236    /// `symmetric_key_ratchet`.
237    ///
238    /// If `active_ratchet` contains root key `R`<sub>`i`</sub> and our own
239    /// ratchet key `T`<sub>`i`</sub>, this is `T`<sub>`i-1`</sub>.
240    ///
241    /// `None` means "unknown", either because this session has been restored
242    /// from a pickle which did not record the parent session key, or because
243    /// this is the first chain in the session.
244    ///
245    /// This is not required to implement the algorithm: it is maintained solely
246    /// for diagnostic output.
247    #[serde(default)]
248    parent_ratchet_key: Option<RemoteRatchetKey>,
249
250    /// The number of times the ratchet has been advanced.
251    ///
252    /// If `active_ratchet` contains root key `R`<sub>`i`</sub>, this is `i`.
253    #[serde(default = "RatchetCount::unknown")]
254    ratchet_count: RatchetCount,
255
256    active_ratchet: Ratchet,
257    symmetric_key_ratchet: ChainKey,
258}
259
260impl ActiveDoubleRatchet {
261    fn advance(
262        &self,
263        ratchet_key: RemoteRatchetKey,
264    ) -> Option<(InactiveDoubleRatchet, ReceiverChain)> {
265        let (root_key, remote_chain) = self.active_ratchet.advance(ratchet_key)?;
266
267        let new_ratchet_count = self.ratchet_count.advance();
268        let ratchet = InactiveDoubleRatchet {
269            root_key,
270            ratchet_key,
271            ratchet_count: new_ratchet_count.clone(),
272        };
273        let receiver_chain = ReceiverChain::new(ratchet_key, remote_chain, new_ratchet_count);
274
275        Some((ratchet, receiver_chain))
276    }
277
278    fn ratchet_key(&self) -> RatchetPublicKey {
279        RatchetPublicKey::from(self.active_ratchet.ratchet_key())
280    }
281
282    fn next_message_key(&mut self) -> MessageKey {
283        self.symmetric_key_ratchet.create_message_key(self.ratchet_key())
284    }
285}

287impl Debug for ActiveDoubleRatchet {
288    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
289        let active_ratchet_public_key: RatchetPublicKey = self.active_ratchet.ratchet_key().into();
290        f.debug_struct("ActiveDoubleRatchet")
291            .field("ratchet_count", &self.ratchet_count)
292            .field("parent_ratchet_key", &self.parent_ratchet_key)
293            .field("ratchet_key", &active_ratchet_public_key)
294            .field("chain_index", &self.symmetric_key_ratchet.index())
295            .finish_non_exhaustive()
296    }
297}

299/// The number of times the ratchet has been advanced, `i`.
300///
301/// This starts at 0 for the first prekey messages from Alice to Bob,
302/// increments to 1 when Bob replies, and then increments each time the
303/// conversation changes direction.
304///
305/// It may be unknown, if the ratchet was restored from a pickle
306/// which didn't track it.
307#[derive(Serialize, Deserialize, Clone, PartialEq, Eq)]
308pub enum RatchetCount {
309    Known(u64),
310    Unknown(()),
311}
312
313impl RatchetCount {
314    pub const fn new() -> RatchetCount {
315        RatchetCount::Known(0)
316    }
317
318    pub const fn unknown() -> RatchetCount {
319        RatchetCount::Unknown(())
320    }
321
322    pub fn advance(&self) -> RatchetCount {
323        match self {
324            RatchetCount::Known(count) => RatchetCount::Known(count + 1),
325            RatchetCount::Unknown(_) => RatchetCount::Unknown(()),
326        }
327    }
328}
329
330impl Debug for RatchetCount {
331    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
332        match self {
333            RatchetCount::Known(count) => write!(f, "{count}"),
334            RatchetCount::Unknown(_) => write!(f, "<unknown>"),
335        }
336    }
337}

351    fn create_session_pair(alice: &Account, bob: &mut Account) -> (Session, Session) {
352        let bob_otks = bob.generate_one_time_keys(1);
353        let bob_otk = bob_otks.created.first().expect("Couldn't get a one-time-key for bob");
354        let bob_identity_key = bob.identity_keys().curve25519;
355        let mut alice_session = alice
356            .create_outbound_session(SessionConfig::version_1(), bob_identity_key, *bob_otk)
357            .unwrap();
358
359        let message = "It's a secret to everybody";
360        let olm_message = alice_session.encrypt(message).unwrap();
361        assert_matches!(olm_message, OlmMessage::PreKey(prekey_message));
362
363        let alice_identity_key = alice.identity_keys().curve25519;
364        let bob_session_creation_result = bob
365            .create_inbound_session(SessionConfig::version_1(), alice_identity_key, &prekey_message)
366            .expect("Unable to create inbound session");
367        assert_eq!(bob_session_creation_result.plaintext, message.as_bytes());
368        (alice_session, bob_session_creation_result.session)
369    }

371    fn assert_active_ratchet(sending_ratchet: &DoubleRatchet) -> &ActiveDoubleRatchet {
372        match &sending_ratchet.inner {
373            DoubleRatchetState::Inactive(_) => panic!("Not an active ratchet"),
374            DoubleRatchetState::Active(s) => s,
375        }
376    }
377
378    fn assert_inactive_ratchet(sending_ratchet: &DoubleRatchet) -> &InactiveDoubleRatchet {
379        match &sending_ratchet.inner {
380            DoubleRatchetState::Active(_) => panic!("Not an inactive ratchet"),
381            DoubleRatchetState::Inactive(s) => s,
382        }
383    }

385    #[test]
386    fn ratchet_counts() {
387        let (mut alice_session, mut bob_session) =
388            create_session_pair(&Account::new(), &mut Account::new());
389
390        // Both ratchets should start with count 0.
391        assert_eq!(
392            assert_active_ratchet(&alice_session.sending_ratchet).ratchet_count,
393            RatchetCount::Known(0)
394        );
395        assert_eq!(
396            assert_inactive_ratchet(&bob_session.sending_ratchet).ratchet_count,
397            RatchetCount::Known(0)
398        );
399
400        // Once Bob replies, the ratchets should bump to 1.
401        let olm_message = bob_session.encrypt("sssh").unwrap();
402        alice_session.decrypt(&olm_message).expect("Alice could not decrypt message from Bob");
403        assert_eq!(
404            assert_inactive_ratchet(&alice_session.sending_ratchet).ratchet_count,
405            RatchetCount::Known(1)
406        );
407        assert_eq!(
408            assert_active_ratchet(&bob_session.sending_ratchet).ratchet_count,
409            RatchetCount::Known(1)
410        );
411
412        // Now Alice replies again.
413        let olm_message = alice_session.encrypt("sssh").unwrap();
414        bob_session.decrypt(&olm_message).expect("Bob could not decrypt message from Alice");
415        assert_eq!(
416            assert_active_ratchet(&alice_session.sending_ratchet).ratchet_count,
417            RatchetCount::Known(2)
418        );
419        assert_eq!(
420            assert_inactive_ratchet(&bob_session.sending_ratchet).ratchet_count,
421            RatchetCount::Known(2)
422        );
423    }

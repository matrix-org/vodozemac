15use hkdf::Hkdf;
16use serde::{Deserialize, Serialize};
17use sha2::Sha256;
18use zeroize::{Zeroize, ZeroizeOnDrop};
19
20use super::{
21    chain_key::{ChainKey, RemoteChainKey},
22    ratchet::{RatchetKey, RemoteRatchetKey},
23};
24
25const ADVANCEMENT_SEED: &[u8; 11] = b"OLM_RATCHET";
26
27/// A root key for one of our own sender chains.
28///
29/// A new root key `R`<sub>`i`</sub> is calculated each time the conversation
30/// changes direction, based on the previous root key `R`<sub>`i-1`</sub> and
31/// the previous and new [ratchet keys](RatchetKey) `T`<sub>`i-1`</sub>,
32/// `T`<sub>`i`</sub>. It is used only to calculate the *next* root key
33/// `R`<sub>`i+1`</sub> and [chain key](ChainKey) `C`<sub>`i+1`</sub>.
34///
35/// This struct holds the root key corresponding to chains where we are the
36/// sender. See also [`RemoteRootKey`].
37#[derive(Clone, Serialize, Deserialize, Zeroize, ZeroizeOnDrop)]
38#[serde(transparent)]
39pub(crate) struct RootKey {
40    pub key: Box<[u8; 32]>,
41}
42
43/// A root key for one of the other side's sender chains.
44///
45/// See [`RootKey`] for information on root keys. This struct holds the root key
46/// corresponding to chains where the other side is the sender.
47#[derive(Clone, Serialize, Deserialize, Zeroize, ZeroizeOnDrop)]
48pub(crate) struct RemoteRootKey {
49    pub key: Box<[u8; 32]>,
50}
51
52fn kdf(
53    root_key: &[u8; 32],
54    ratchet_key: &RatchetKey,
55    remote_ratchet_key: &RemoteRatchetKey,
56) -> Box<[u8; 64]> {
57    let shared_secret = ratchet_key.diffie_hellman(remote_ratchet_key);
58    let hkdf: Hkdf<Sha256> = Hkdf::new(Some(root_key.as_ref()), shared_secret.as_bytes());
59    let mut output = Box::new([0u8; 64]);
60
61    #[allow(clippy::expect_used)]
62    hkdf.expand(ADVANCEMENT_SEED, output.as_mut_slice())
63        .expect("We should be able to expand the shared secret.");
64
65    output
66}
67
68impl RemoteRootKey {
69    pub(super) const fn new(bytes: Box<[u8; 32]>) -> Self {
70        Self { key: bytes }
71    }
72
73    pub(super) fn advance(
74        &self,
75        remote_ratchet_key: &RemoteRatchetKey,
76    ) -> (RootKey, ChainKey, RatchetKey) {
77        let ratchet_key = RatchetKey::new();
78        let output = kdf(&self.key, &ratchet_key, remote_ratchet_key);
79
80        let mut chain_key = Box::new([0u8; 32]);
81        let mut root_key = Box::new([0u8; 32]);
82
83        chain_key.copy_from_slice(&output[32..]);
84        root_key.copy_from_slice(&output[..32]);
85
86        let chain_key = ChainKey::new(chain_key);
87        let root_key = RootKey::new(root_key);
88
89        (root_key, chain_key, ratchet_key)
90    }
91}
92
93impl RootKey {
94    pub(super) const fn new(bytes: Box<[u8; 32]>) -> Self {
95        Self { key: bytes }
96    }
97
98    pub(super) fn advance(
99        &self,
100        old_ratchet_key: &RatchetKey,
101        remote_ratchet_key: &RemoteRatchetKey,
102    ) -> (RemoteRootKey, RemoteChainKey) {
103        let output = kdf(&self.key, old_ratchet_key, remote_ratchet_key);
104
105        let mut chain_key = Box::new([0u8; 32]);
106        let mut root_key = Box::new([0u8; 32]);
107
108        root_key.copy_from_slice(&output[..32]);
109        chain_key.copy_from_slice(&output[32..]);
110
111        let root_key = RemoteRootKey::new(root_key);
112        let chain_key = RemoteChainKey::new(chain_key);
113
114        (root_key, chain_key)
115    }
116}

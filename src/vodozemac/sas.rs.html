15//! User-friendly key verification using short authentication strings (SAS).
16//!
17//! The verification process is heavily inspired by Phil Zimmermann’s [ZRTP]
18//! key agreement handshake. A core part of key agreement in [ZRTP] is the
19//! *hash commitment*: the party that begins the key sharing process sends
20//! a *hash* of their part of the Diffie-Hellman exchange but does not send the
21//! part itself exchange until they had received the other party’s part.
22//!
23//! The verification process can be used to verify the Ed25519 identity key of
24//! an [`Account`].
25//!
26//! # Examples
27//!
28//! ```rust
29//! use vodozemac::sas::Sas;
30//! # use anyhow::Result;
31//! # fn main() -> Result<()> {
32//! let alice = Sas::new();
33//! let bob = Sas::new();
34//!
35//! let bob_public_key = bob.public_key();
36//!
37//! let bob = bob.diffie_hellman(alice.public_key())?;
38//! let alice = alice.diffie_hellman(bob_public_key)?;
39//!
40//! let alice_bytes = alice.bytes("AGREED_INFO");
41//! let bob_bytes = bob.bytes("AGREED_INFO");
42//!
43//! let alice_emojis = alice_bytes.emoji_indices();
44//! let bob_emojis = bob_bytes.emoji_indices();
45//!
46//! assert_eq!(alice_emojis, bob_emojis);
47//! # Ok(())
48//! # }
49//! ```
50//!
51//! [`Account`]: crate::olm::Account
52//! [ZRTP]: https://tools.ietf.org/html/rfc6189#section-4.4.1

68/// The output type for the SAS MAC calculation.
69pub struct Mac(Vec<u8>);

71impl Mac {
72    /// Convert the MAC to a base64 encoded string.
73    pub fn to_base64(&self) -> String {
74        base64_encode(&self.0)
75    }
76
77    /// Get the byte slice of the MAC.
78    pub fn as_bytes(&self) -> &[u8] {
79        &self.0
80    }
81
82    /// Create a new `Mac` object from a byte slice.
83    pub fn from_slice(bytes: &[u8]) -> Self {
84        Self(bytes.to_vec())
85    }
86
87    /// Create a new `Mac` object from a base64 encoded string.
88    pub fn from_base64(mac: &str) -> Result<Self, base64::DecodeError> {
89        let bytes = base64_decode(mac)?;
90
91        Ok(Self(bytes))
92    }
93}

137/// Bytes generated from an shared secret that can be used as the short auth
138/// string.
139#[derive(Debug, Clone, PartialEq, Eq)]
140pub struct SasBytes {
141    bytes: [u8; 6],
142}
143
144impl SasBytes {
145    /// Get the index of 7 emojis that can be presented to users to perform the
146    /// key verification
147    ///
148    /// The table that maps the index to an emoji can be found in the [spec].
149    ///
150    /// [spec]: https://spec.matrix.org/unstable/client-server-api/#sas-method-emoji
151    pub fn emoji_indices(&self) -> [u8; 7] {
152        Self::bytes_to_emoji_index(&self.bytes)
153    }
154
155    /// Get the three decimal numbers that can be presented to users to perform
156    /// the key verification, as described in the [spec]
157    ///
158    /// [spec]: https://spec.matrix.org/unstable/client-server-api/#sas-method-emoji
159    pub fn decimals(&self) -> (u16, u16, u16) {
160        Self::bytes_to_decimal(&self.bytes)
161    }
162
163    /// Get the raw bytes of the short auth string that can be converted to an
164    /// emoji, or decimal representation.
165    pub const fn as_bytes(&self) -> &[u8; 6] {
166        &self.bytes
167    }
168
169    /// Split the first 42 bits of our 6 bytes into 7 groups of 6 bits. The 7
170    /// groups of 6 bits represent an emoji index from the [spec].
171    ///
172    /// [spec]: https://spec.matrix.org/unstable/client-server-api/#sas-method-emoji
173    fn bytes_to_emoji_index(bytes: &[u8; 6]) -> [u8; 7] {
174        let bytes: Vec<u64> = bytes.iter().map(|b| *b as u64).collect();
175        // Join the 6 bytes into one 64 bit unsigned int. This u64 will contain 48
176        // bits from our 6 bytes.
177        let mut num: u64 = bytes[0] << 40;
178        num += bytes[1] << 32;
179        num += bytes[2] << 24;
180        num += bytes[3] << 16;
181        num += bytes[4] << 8;
182        num += bytes[5];
183
184        // Take the top 42 bits of our 48 bits from the u64 and convert each 6 bits
185        // into a 6 bit number.
186        [
187            ((num >> 42) & 63) as u8,
188            ((num >> 36) & 63) as u8,
189            ((num >> 30) & 63) as u8,
190            ((num >> 24) & 63) as u8,
191            ((num >> 18) & 63) as u8,
192            ((num >> 12) & 63) as u8,
193            ((num >> 6) & 63) as u8,
194        ]
195    }
196
197    /// Convert the given bytes into three decimals. The 6th byte is ignored,
198    /// it's used for the emoji index conversion.
199    fn bytes_to_decimal(bytes: &[u8; 6]) -> (u16, u16, u16) {
200        let bytes: Vec<u16> = bytes.iter().map(|b| *b as u16).collect();
201
202        // This bitwise operation is taken from the [spec]
203        // [spec]: https://matrix.org/docs/spec/client_server/latest#sas-method-decimal
204        let first = (bytes[0] << 5) | (bytes[1] >> 3);
205        let second = ((bytes[1] & 0x7) << 10) | (bytes[2] << 2) | (bytes[3] >> 6);
206        let third = ((bytes[3] & 0x3F) << 7) | (bytes[4] >> 1);
207
208        (first + 1000, second + 1000, third + 1000)
209    }
210}

218impl Sas {
219    /// Create a new random verification object
220    ///
221    /// This creates an ephemeral curve25519 keypair that can be used to
222    /// establish a shared secret.
223    pub fn new() -> Self {
224        let rng = thread_rng();
225
226        let secret_key = EphemeralSecret::random_from_rng(rng);
227        let public_key = Curve25519PublicKey::from(&secret_key);
228
229        Self { secret_key, public_key }
230    }
231
232    /// Get the public key that can be used to establish a shared secret.
233    pub const fn public_key(&self) -> Curve25519PublicKey {
234        self.public_key
235    }
236
237    /// Establishes a SAS secret by performing a DH handshake with another
238    /// public key.
239    ///
240    /// Returns an [`EstablishedSas`] object which can be used to generate
241    /// [`SasBytes`] if the given public key was valid, otherwise `None`.
242    pub fn diffie_hellman(
243        self,
244        their_public_key: Curve25519PublicKey,
245    ) -> Result<EstablishedSas, KeyError> {
246        let shared_secret = self.secret_key.diffie_hellman(&their_public_key.inner);
247
248        if shared_secret.was_contributory() {
249            Ok(EstablishedSas { shared_secret, our_public_key: self.public_key, their_public_key })
250        } else {
251            Err(KeyError::NonContributoryKey)
252        }
253    }
254
255    /// Establishes a SAS secret by performing a DH handshake with another
256    /// public key in "raw", base64-encoded form.
257    ///
258    /// Returns an [`EstablishedSas`] object which can be used to generate
259    /// [`SasBytes`] if the received public key is valid, otherwise `None`.
260    pub fn diffie_hellman_with_raw(
261        self,
262        other_public_key: &str,
263    ) -> Result<EstablishedSas, KeyError> {
264        let other_public_key = Curve25519PublicKey::from_base64(other_public_key)?;
265        self.diffie_hellman(other_public_key)
266    }
267}

269impl EstablishedSas {
270    /// Generate [`SasBytes`] using HKDF with the shared secret as the input key
271    /// material.
272    ///
273    /// The info string should be agreed upon beforehand, both parties need to
274    /// use the same info string.
275    pub fn bytes(&self, info: &str) -> SasBytes {
276        let mut bytes = [0u8; 6];
277        #[allow(clippy::expect_used)]
278        let byte_vec =
279            self.bytes_raw(info, 6).expect("HKDF should always be able to generate 6 bytes");
280
281        bytes.copy_from_slice(&byte_vec);
282
283        SasBytes { bytes }
284    }
285
286    /// Generate the given number of bytes using HKDF with the shared secret
287    /// as the input key material.
288    ///
289    /// The info string should be agreed upon beforehand, both parties need to
290    /// use the same info string.
291    ///
292    /// The number of bytes we can generate is limited, we can generate up to
293    /// 32 * 255 bytes. The function will not fail if the given count is smaller
294    /// than the limit.
295    pub fn bytes_raw(&self, info: &str, count: usize) -> Result<Vec<u8>, InvalidCount> {
296        let mut output = vec![0u8; count];
297        let hkdf = self.get_hkdf();
298
299        hkdf.expand(info.as_bytes(), &mut output[0..count]).map_err(|_| InvalidCount)?;
300
301        Ok(output)
302    }
303
304    /// Calculate a MAC for the given input using the info string as additional
305    /// data.
306    ///
307    ///
308    /// This should be used to calculate a MAC of the ed25519 identity key of an
309    /// [`Account`]
310    ///
311    /// The MAC is returned as a base64 encoded string.
312    ///
313    /// [`Account`]: crate::olm::Account
314    pub fn calculate_mac(&self, input: &str, info: &str) -> Mac {
315        let mut mac = self.get_mac(info);
316
317        mac.update(input.as_ref());
318
319        Mac(mac.finalize().into_bytes().to_vec())
320    }
321
322    /// Calculate a MAC for the given input using the info string as additional
323    /// data, the MAC is returned as an invalid base64 encoded string.
324    ///
325    /// **Warning**: This method should never be used unless you require libolm
326    /// compatibility. Libolm used to incorrectly encode their MAC because the
327    /// input buffer was reused as the output buffer. This method replicates the
328    /// buggy behaviour.
329    #[cfg(feature = "libolm-compat")]
330    pub fn calculate_mac_invalid_base64(&self, input: &str, info: &str) -> String {
331        // First calculate the MAC as usual.
332        let mac = self.calculate_mac(input, info);
333
334        // Since the input buffer is reused as an output buffer, and base64
335        // operates on 3 input bytes to generate 4 output bytes, the input
336        // buffer gets overrun by the output.
337        //
338        // Only 6 bytes of the MAC get to be used before the output overwrites
339        // the input.
340
341        // All three bytes of the first input chunk are used successfully.
342        let mut out = base64_encode(&mac.as_bytes()[0..3]);
343
344        // For the next input chunk, only two bytes are sourced from the actual
345        // MAC, since the first byte gets overwritten by the output.
346        let mut bytes_from_mac = 2;
347
348        // Subsequent input chunks get progressively more overwritten by the
349        // output, so that after two iterations, none of the original input
350        // bytes remain.
351        for i in (6..10).step_by(3) {
352            let from_mac = &mac.as_bytes()[i - bytes_from_mac..i];
353            let from_out = &out.as_bytes()[out.len() - (3 - bytes_from_mac)..];
354
355            let bytes = [from_out, from_mac].concat();
356            let encoded = base64_encode(bytes);
357            bytes_from_mac -= 1;
358
359            out = out + &encoded;
360        }
361
362        // At this point, the rest of our input will be completely sourced from
363        // the previous output. The MAC has a size of 32, so we abort before we
364        // get to the remainder calculation.
365        for i in (9..30).step_by(3) {
366            let next = &out.as_bytes()[i..i + 3];
367            let next_four = base64_encode(next);
368            out = out + &next_four;
369        }
370
371        // Finally, use the remainder to get the last 3 bytes of output. No
372        // padding is used.
373        let next = &out.as_bytes()[30..32];
374        let next = base64_encode(next);
375
376        out + &next
377    }
378
379    /// Verify a MAC that was previously created using the
380    /// [`EstablishedSas::calculate_mac()`] method.
381    ///
382    /// Users should calculate a MAC and send it to the other side, they should
383    /// then verify each other's MAC using this method.
384    pub fn verify_mac(&self, input: &str, info: &str, tag: &Mac) -> Result<(), SasError> {
385        let mut mac = self.get_mac(info);
386        mac.update(input.as_bytes());
387
388        Ok(mac.verify_slice(&tag.0)?)
389    }
390
391    /// Get the public key that was created by us, that was used to establish
392    /// the shared secret.
393    pub const fn our_public_key(&self) -> Curve25519PublicKey {
394        self.our_public_key
395    }
396
397    /// Get the public key that was created by the other party, that was used to
398    /// establish the shared secret.
399    pub const fn their_public_key(&self) -> Curve25519PublicKey {
400        self.their_public_key
401    }
402
403    fn get_hkdf(&self) -> Hkdf<Sha256> {
404        Hkdf::new(None, self.shared_secret.as_bytes())
405    }
406
407    fn get_mac_key(&self, info: &str) -> HmacSha256Key {
408        let mut mac_key = Box::new([0u8; 32]);
409        let hkdf = self.get_hkdf();
410
411        #[allow(clippy::expect_used)]
412        hkdf.expand(info.as_bytes(), mac_key.as_mut_slice())
413            .expect("We should be able to expand the shared SAS secret into a MAC key");
414
415        mac_key
416    }
417
418    fn get_mac(&self, info: &str) -> Hmac<Sha256> {
419        let mac_key = self.get_mac_key(info);
420
421        #[allow(clippy::expect_used)]
422        Hmac::<Sha256>::new_from_slice(mac_key.as_slice())
423            .expect("We should be able to create a HMAC object from a 32-byte slice")
424    }
